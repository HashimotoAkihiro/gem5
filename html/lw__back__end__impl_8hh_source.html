00051 template <class Impl>
00052 int
00053 LWBackEnd<Impl>::wakeDependents(DynInstPtr &inst, bool memory_deps)
00054 {
00055     assert(!inst->isSquashed());
00056     std::vector<DynInstPtr> &dependents = memory_deps ? inst->getMemDeps() :
00057         inst->getDependents();
00058     int num_outputs = dependents.size();
00059 
00060     DPRINTF(BE, "Waking instruction [sn:%lli] dependents in IQ\n", inst->seqNum);
00061 
00062     for (int i = 0; i < num_outputs; i++) {
00063         DynInstPtr dep_inst = dependents[i];
00064         if (!memory_deps) {
00065             dep_inst->markSrcRegReady();
00066         } else {
00067             if (!dep_inst->isSquashed())
00068                 dep_inst->markMemInstReady(inst.get());
00069         }
00070 
00071         DPRINTF(BE, "Marking source reg ready [sn:%lli] in IQ\n", dep_inst->seqNum);
00072 
00073         if (dep_inst->readyToIssue() && dep_inst->isInROB() &&
00074             !dep_inst->isNonSpeculative() && !dep_inst->isStoreConditional() &&
00075             dep_inst->memDepReady() && !dep_inst->isMemBarrier() &&
00076             !dep_inst->isWriteBarrier()) {
00077             DPRINTF(BE, "Adding instruction to exeList [sn:%lli]\n",
00078                     dep_inst->seqNum);
00079             exeList.push(dep_inst);
00080             if (dep_inst->iqItValid) {
00081                 DPRINTF(BE, "Removing instruction from waiting list\n");
00082                 waitingList.erase(dep_inst->iqIt);
00083                 waitingInsts--;
00084                 dep_inst->iqItValid = false;
00085                 assert(waitingInsts >= 0);
00086             }
00087             if (dep_inst->isMemRef()) {
00088                 removeWaitingMemOp(dep_inst);
00089                 DPRINTF(BE, "Issued a waiting mem op [sn:%lli]\n",
00090                         dep_inst->seqNum);
00091             }
00092         }
00093     }
00094     return num_outputs;
00095 }

00521 template <class Impl>
00522 void
00523 LWBackEnd<Impl>::checkInterrupts()
00524 {
00525     if (cpu->checkInterrupts(tc) && !trapSquash && !tcSquash) {
00526         frontEnd->interruptPending = true;
00527         if (robEmpty() && !LSQ.hasStoresToWB()) {
00528             // Will need to squash all instructions currently in flight and have
00529             // the interrupt handler restart at the last non-committed inst.
00530             // Most of that can be handled through the trap() function.  The
00531             // processInterrupts() function really just checks for interrupts
00532             // and then calls trap() if there is an interrupt present.
00533 
00534             // Not sure which thread should be the one to interrupt.  For now
00535             // always do thread 0.
00536             assert(!thread->noSquashFromTC);
00537             thread->noSquashFromTC = true;
00538 
00539             // CPU will handle implementation of the interrupt.
00540             cpu->processInterrupts();
00541 
00542             // Now squash or record that I need to squash this cycle.
00543             commitStatus = TrapPending;
00544 
00545             // Exit state update mode to avoid accidental updating.
00546             thread->noSquashFromTC = false;
00547 
00548             // Generate trap squash event.
00549             generateTrapEvent();
00550 
00551             DPRINTF(BE, "Interrupt detected.\n");
00552         } else {
00553             DPRINTF(BE, "Interrupt must wait for ROB to drain.\n");
00554         }
00555     }
00556 }

00584 {
00585     DPRINTF(BE, "Ticking back end\n");
00586 
00587     // Read in any done instruction information and update the IQ or LSQ.
00588     updateStructures();
00589 
00590     if (switchPending && robEmpty() && !LSQ.hasStoresToWB()) {
00591         cpu->signalSwitched();
00592         return;
00593     }
00594 
00595     readyInstsForCommit();
00596 
00597     numInstsToWB.advance();
00598 
00599     ROBCount[0]+= numInsts;
00600 
00601     wbCycle = 0;
00602 
00603     checkInterrupts();
00604 
00605     if (trapSquash) {
00606         assert(!tcSquash);
00607         squashFromTrap();
00608     } else if (tcSquash) {
00609         squashFromTC();
00610     }
00611 
00612     if (dispatchStatus != Blocked) {
00613         dispatchInsts();
00614     } else {
00615         checkDispatchStatus();
00616     }
00617 
00618     if (commitStatus != TrapPending) {
00619         executeInsts();
00620 
00621         commitInsts();
00622     }
00623 
00624     LSQ.writebackStores();
00625 
00626     DPRINTF(BE, "Waiting insts: %i, mem ops: %i, ROB entries in use: %i, "
00627             "LSQ loads: %i, LSQ stores: %i\n",
00628             waitingInsts, numWaitingMemOps, numInsts,
00629             LSQ.numLoads(), LSQ.numStores());
00630 
00631 #ifdef DEBUG
00632     assert(numInsts == instList.size());
00633     assert(waitingInsts == waitingList.size());
00634     assert(numWaitingMemOps == waitingMemOps.size());
00635     assert(!switchedOut);
00636 #endif
00637 }

00666 template <class Impl>
00667 void
00668 LWBackEnd<Impl>::dispatchInsts()
00669 {
00670     DPRINTF(BE, "Trying to dispatch instructions.\n");
00671 
00672     while (numInsts < numROBEntries &&
00673            numWaitingMemOps < maxOutstandingMemOps) {
00674         // Get instruction from front of time buffer
00675         if (lsqLimits && LSQ.isFull()) {
00676             break;
00677         }
00678 
00679         DynInstPtr inst = frontEnd->getInst();
00680         if (!inst) {
00681             break;
00682         } else if (inst->isSquashed()) {
00683             continue;
00684         }
00685 
00686         ++numInsts;
00687         instList.push_front(inst);
00688 
00689         inst->setInROB();
00690 
00691         DPRINTF(BE, "Dispatching instruction [sn:%lli] PC:%#x\n",
00692                 inst->seqNum, inst->readPC());
00693 
00694         for (int i = 0; i < inst->numDestRegs(); ++i)
00695             renameTable[inst->destRegIdx(i)] = inst;
00696
