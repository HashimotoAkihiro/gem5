00410         bool aarch64;
00411 
00413         bool largeGrain;
00414 
00416         int grainSize;
00417 
00419         EntryType type() const
00420         {
00421             switch (bits(data, 1, 0)) {
00422               case 0x1:
00423                 // In AArch64 blocks are not allowed at L0 for the 4 KB granule
00424                 // and at L1 for the 64 KB granule
00425                 if (largeGrain)
00426                     return lookupLevel == L2 ? Block : Invalid;
00427                 return lookupLevel == L0 || lookupLevel == L3 ? Invalid : Block;
00428               case 0x3:
00429                 return lookupLevel == L3 ? Page : Table;
00430               default:
00431                 return Invalid;
00432             }
00433         }
00434 
00436         uint8_t offsetBits() const
00437         {
00438             assert(type() == Block || type() == Page);
00439             if (largeGrain) {
00440                 if (type() == Block)
00441                     return 29 /* 512 MB */;
00442                 return 16 /* 64 KB */;  // type() == Page
00443             } else {
00444                 if (type() == Block)
00445                     return lookupLevel == L1 ? 30 /* 1 GB */ : 21 /* 2 MB */;
00446                 return 12 /* 4 KB */;  // type() == Page
00447             }
00448         }
00449 
00451         Addr pfn() const
00452         {
00453             if (aarch64)
00454                 return bits(data, 47, offsetBits());
00455             return bits(data, 39, offsetBits());
00456         }

00460         {

00486         Addr nextDescAddr(Addr va) const
00487         {
00488             assert(type() == Table);
00489             Addr pa = 0;
00490             if (aarch64) {
00491                 int stride = grainSize - 3;
00492                 int va_lo = stride * (3 - (lookupLevel + 1)) + grainSize;
00493                 int va_hi = va_lo + stride - 1;
00494                 pa = nextTableAddr() | (bits(va, va_hi, va_lo) << 3);
00495             } else {
00496                 if (lookupLevel == L1)
00497                     pa = nextTableAddr() | (bits(va, 29, 21) << 3);
00498                 else  // lookupLevel == L2
00499                     pa = nextTableAddr() | (bits(va, 20, 12) << 3);
00500             }
