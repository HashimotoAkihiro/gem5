00133 bool
00134 BPredUnit::predict(StaticInstPtr &inst, const InstSeqNum &seqNum,
00135                    TheISA::PCState &pc, ThreadID tid)
00136 {
00137     // See if branch predictor predicts taken.
00138     // If so, get its target addr either from the BTB or the RAS.
00139     // Save off record of branch stuff so the RAS can be fixed
00140     // up once it's done.
00141 
00142     bool pred_taken = false;
00143     TheISA::PCState target = pc;
00144 
00145     ++lookups;
00146 
00147     void *bp_history = NULL;
00148 
00149     if (inst->isUncondCtrl()) {
00150         DPRINTF(Branch, "[tid:%i]: Unconditional control.\n", tid);
00151         pred_taken = true;
00152         // Tell the BP there was an unconditional branch.
00153         uncondBranch(bp_history);
00154     } else {
00155         ++condPredicted;
00156         pred_taken = lookup(pc.instAddr(), bp_history);
00157 
00158         DPRINTF(Branch, "[tid:%i]: [sn:%i] Branch predictor"
00159                 " predicted %i for PC %s\n", tid, seqNum,  pred_taken, pc);
00160     }
00161 
00162     DPRINTF(Branch, "[tid:%i]: [sn:%i] Creating prediction history "
00163             "for PC %s\n", tid, seqNum, pc);
00164 
00165     PredictorHistory predict_record(seqNum, pc.instAddr(),
00166                                     pred_taken, bp_history, tid);
00167 
00168     // Now lookup in the BTB or RAS.
00169     if (pred_taken) {
00170         if (inst->isReturn()) {
00171             ++usedRAS;
00172             predict_record.wasReturn = true;
00173             // If it's a function return call, then look up the address
00174             // in the RAS.
00175             TheISA::PCState rasTop = RAS[tid].top();
00176             target = TheISA::buildRetPC(pc, rasTop);
00177 
00178             // Record the top entry of the RAS, and its index.
00179             predict_record.usedRAS = true;
00180             predict_record.RASIndex = RAS[tid].topIdx();
00181             predict_record.RASTarget = rasTop;
00182 
00183             RAS[tid].pop();
00184 
00185             DPRINTF(Branch, "[tid:%i]: Instruction %s is a return, "
00186                     "RAS predicted target: %s, RAS index: %i.\n",
00187                     tid, pc, target, predict_record.RASIndex);
00188         } else {
00189             ++BTBLookups;
00190 
00191             if (inst->isCall()) {
00192                 RAS[tid].push(pc);
00193                 predict_record.pushedRAS = true;
00194 
00195                 // Record that it was a call so that the top RAS entry can
00196                 // be popped off if the speculation is incorrect.
00197                 predict_record.wasCall = true;
00198 
00199                 DPRINTF(Branch, "[tid:%i]: Instruction %s was a "
00200                         "call, adding %s to the RAS index: %i.\n",
00201                         tid, pc, pc, RAS[tid].topIdx());
00202             }
00203 
00204             if (BTB.valid(pc.instAddr(), tid)) {
00205                 ++BTBHits;
00206 
00207                 // If it's not a return, use the BTB to get the target addr.
00208                 target = BTB.lookup(pc.instAddr(), tid);
00209 
00210                 DPRINTF(Branch, "[tid:%i]: Instruction %s predicted"
00211                         " target is %s.\n", tid, pc, target);
00212 
00213             } else {
00214                 DPRINTF(Branch, "[tid:%i]: BTB doesn't have a "
00215                         "valid entry.\n",tid);
00216                 pred_taken = false;
00217                 // The Direction of the branch predictor is altered because the
00218                 // BTB did not have an entry
00219                 // The predictor needs to be updated accordingly
00220                 if (!inst->isCall() && !inst->isReturn()) {
00221                       btbUpdate(pc.instAddr(), bp_history);
00222                       DPRINTF(Branch, "[tid:%i]:[sn:%i] btbUpdate"
00223                               " called for %s\n", tid, seqNum, pc);
00224                 } else if (inst->isCall() && !inst->isUncondCtrl()) {
00225                       RAS[tid].pop();
00226                       predict_record.pushedRAS = false;
00227                 }
00228                 TheISA::advancePC(target, inst);
00229             }
00230         }
00231     } else {
00232         if (inst->isReturn()) {
00233            predict_record.wasReturn = true;
00234         }
00235         TheISA::advancePC(target, inst);
00236     }
00237 
00238     pc = target;
00239 
00240     predHist[tid].push_front(predict_record);
00241 
00242     DPRINTF(Branch, "[tid:%i]: [sn:%i]: History entry added."
00243             "predHist.size(): %i\n", tid, seqNum, predHist[tid].size());
00244 
00245     return pred_taken;
00246 }

00280         }
00281     } else {
00282         ++condPredicted;

00299

00301             predict_record.usedRAS = true;
00302             predict_record.RASIndex = RAS[tid].topIdx();
00303             predict_record.RASTarget = rasTop;
00304 
00305             assert(predict_record.RASIndex < 16);
00306 
00307             RAS[tid].pop();
00308 
00309             DPRINTF(Branch, "[tid:%i]: Instruction %s is a return, "
00310                     "RAS predicted target: %s, RAS index: %i.\n",
00311                     tid, instPC, target,
00312                     predict_record.RASIndex);
00313         } else {
00314             ++BTBLookups;
00315 
00316             if (inst->isCall()) {
00317 
00318                 RAS[tid].push(instPC);
00319                 predict_record.pushedRAS = true;
00320 
00321                 // Record that it was a call so that the top RAS entry can

00327                         tid, instPC, predPC,
00328                         RAS[tid].topIdx());
00329             }
00330 
00331             if (inst->isCall() &&
00332                 inst->isUncondCtrl() &&
00333                 inst->isDirectCtrl()) {
00334                 target = inst->branchTarget(instPC);
00335             } else if (BTB.valid(predPC.instAddr(), asid)) {
00336                 ++BTBHits;
00337 
00338                 // If it's not a return, use the BTB to get the target addr.
00339                 target = BTB.lookup(predPC.instAddr(), asid);
00340 
00341                 DPRINTF(Branch, "[tid:%i]: [asid:%i] Instruction %s "

00345                 DPRINTF(Branch, "[tid:%i]: BTB doesn't have a "
00346                         "valid entry, predicting false.\n",tid);
00347                 pred_taken = false;
00348             }
00349         }
00350     }
00351 
00352     if (pred_taken) {
00353         // Set the PC and the instruction's predicted target.
00354         predPC = target;

00356     DPRINTF(Branch, "[tid:%i]: [sn:%i]: Setting Predicted PC to %s.\n",

00360

00368 void
00369 BPredUnit::update(const InstSeqNum &done_sn, ThreadID tid)
00370 {
00371     DPRINTF(Branch, "[tid:%i]: Committing branches until "
00372             "[sn:%lli].\n", tid, done_sn);
00373 
00374     while (!predHist[tid].empty() &&
00375            predHist[tid].back().seqNum <= done_sn) {
00376         // Update the branch predictor with the correct results.
00377         update(predHist[tid].back().pc, predHist[tid].back().predTaken,
00378                predHist[tid].back().bpHistory, false);
00379 
00380         predHist[tid].pop_back();
00381     }

00384 void
00385 BPredUnit::squash(const InstSeqNum &squashed_sn, ThreadID tid)
00386 {
00387     History &pred_hist = predHist[tid];
00388 
00389     while (!pred_hist.empty() &&
00390            pred_hist.front().seqNum > squashed_sn) {
00391         if (pred_hist.front().usedRAS) {
00392             DPRINTF(Branch, "[tid:%i]: Restoring top of RAS to: %i,"
00393                     " target: %s.\n", tid,
00394                     pred_hist.front().RASIndex, pred_hist.front().RASTarget);
00395 
00396             RAS[tid].restore(pred_hist.front().RASIndex,
00397                              pred_hist.front().RASTarget);
00398         } else if(pred_hist.front().wasCall && pred_hist.front().pushedRAS) {
00399              // Was a call but predicated false. Pop RAS here
00400              DPRINTF(Branch, "[tid: %i] Squashing"
00401                      "  Call [sn:%i] PC: %s Popping RAS\n", tid,
00402                      pred_hist.front().seqNum, pred_hist.front().pc);
00403              RAS[tid].pop();
00404         }
00405 
00406         // This call should delete the bpHistory.
00407         squash(pred_hist.front().bpHistory);
00408 
00409         DPRINTF(Branch, "[tid:%i]: Removing history for [sn:%i] "
00410                 "PC %s.\n", tid, pred_hist.front().seqNum,
00411                 pred_hist.front().pc);
00412 
00413         pred_hist.pop_front();
00414 
00415         DPRINTF(Branch, "[tid:%i]: predHist.size(): %i\n",
00416                 tid, predHist[tid].size());
00417     }
00418 }

00420 void
00421 BPredUnit::squash(const InstSeqNum &squashed_sn,
00422                   const TheISA::PCState &corrTarget,
00423                   bool actually_taken, ThreadID tid)
00424 {
00425     // Now that we know that a branch was mispredicted, we need to undo
00426     // all the branches that have been seen up until this branch and
00427     // fix up everything.
00428     // NOTE: This should be call conceivably in 2 scenarios:
00429     // (1) After an branch is executed, it updates its status in the ROB
00430     //     The commit stage then checks the ROB update and sends a signal to
00431     //     the fetch stage to squash history after the mispredict
00432     // (2) In the decode stage, you can find out early if a unconditional
00433     //     PC-relative, branch was predicted incorrectly. If so, a signal
00434     //     to the fetch stage is sent to squash history after the mispredict
00435 
00436     History &pred_hist = predHist[tid];
00437 
00438     ++condIncorrect;
00439 
00440     DPRINTF(Branch, "[tid:%i]: Squashing from sequence number %i, "
00441             "setting target to %s.\n", tid, squashed_sn, corrTarget);
00442 
00443     // Squash All Branches AFTER this mispredicted branch
00444     squash(squashed_sn, tid);
00445 
00446     // If there's a squash due to a syscall, there may not be an entry
00447     // corresponding to the squash.  In that case, don't bother trying to
00448     // fix up the entry.
00449     if (!pred_hist.empty()) {
00450 
00451         HistoryIt hist_it = pred_hist.begin();
00452         //HistoryIt hist_it = find(pred_hist.begin(), pred_hist.end(),
00453         //                       squashed_sn);
00454 
00455         //assert(hist_it != pred_hist.end());
00456         if (pred_hist.front().seqNum != squashed_sn) {
00457             DPRINTF(Branch, "Front sn %i != Squash sn %i\n",
00458                     pred_hist.front().seqNum, squashed_sn);
00459 
00460             assert(pred_hist.front().seqNum == squashed_sn);
00461         }
00462 
00463 
00464         if ((*hist_it).usedRAS) {
00465             ++RASIncorrect;
00466         }
00467 
00468         update((*hist_it).pc, actually_taken,
00469                pred_hist.front().bpHistory, true);
00470         if (actually_taken) {
00471             if (hist_it->wasReturn && !hist_it->usedRAS) {
00472                  DPRINTF(Branch, "[tid: %i] Incorrectly predicted"
00473                          "  return [sn:%i] PC: %s\n", tid, hist_it->seqNum,
00474                          hist_it->pc);
00475                  RAS[tid].pop();
00476             }
00477 
00478             DPRINTF(Branch,"[tid: %i] BTB Update called for [sn:%i]"
00479                     " PC: %s\n", tid,hist_it->seqNum, hist_it->pc);
00480 
00481             BTB.update((*hist_it).pc, corrTarget, tid);
00482 
00483         } else {
00484            //Actually not Taken
00485            if (hist_it->usedRAS) {
00486                 DPRINTF(Branch,"[tid: %i] Incorrectly predicted"
00487                         "  return [sn:%i] PC: %s Restoring RAS\n", tid,
00488                         hist_it->seqNum, hist_it->pc);
00489                 DPRINTF(Branch, "[tid:%i]: Restoring top of RAS"
00490                         " to: %i, target: %s.\n", tid,
00491                         hist_it->RASIndex, hist_it->RASTarget);
00492                 RAS[tid].restore(hist_it->RASIndex, hist_it->RASTarget);
00493 
00494            } else if (hist_it->wasCall && hist_it->pushedRAS) {
00495                  //Was a Call but predicated false. Pop RAS here
00496                  DPRINTF(Branch, "[tid: %i] Incorrectly predicted"
00497                          "  Call [sn:%i] PC: %s Popping RAS\n", tid,
00498                          hist_it->seqNum, hist_it->pc);
00499                  RAS[tid].pop();
00500            }
00501         }
00502         DPRINTF(Branch, "[tid:%i]: Removing history for [sn:%i]"
00503                 " PC %s  Actually Taken: %i\n", tid, hist_it->seqNum,
00504                 hist_it->pc, actually_taken);
00505 
00506         pred_hist.erase(hist_it);
00507 
00508         DPRINTF(Branch, "[tid:%i]: predHist.size(): %i\n", tid,
00509                                          predHist[tid].size());
00510     } else {
00511         DPRINTF(Branch, "[tid:%i]: [sn:%i] pred_hist empty, can't "
00512                 "update.\n", tid, squashed_sn);
00513     }
