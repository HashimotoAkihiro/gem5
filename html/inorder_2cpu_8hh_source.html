00392     static const uint8_t INST_LOAD                          = 25;
00393     static const uint8_t INST_STORE                         = 24;
00394     static const uint8_t INST_CONTROL                       = 23;
00395     static const uint8_t INST_NONSPEC                       = 22;
00396     static const uint8_t INST_DEST_REGS                     = 18;
00397     static const uint8_t INST_SRC_REGS                      = 14;
00398     static const uint8_t INST_SPLIT_DATA                    = 13;
00399 
00400     inline SkedID genSkedID(DynInstPtr inst)
00401     {
00402         SkedID id = 0;
00403         id = (inst->opClass() << INST_OPCLASS) |
00404             (inst->isLoad() << INST_LOAD) |
00405             (inst->isStore() << INST_STORE) |
00406             (inst->isControl() << INST_CONTROL) |
00407             (inst->isNonSpeculative() << INST_NONSPEC) |
00408             (inst->numDestRegs() << INST_DEST_REGS) |
00409             (inst->numSrcRegs() << INST_SRC_REGS) |
00410             (inst->splitInst << INST_SPLIT_DATA);
00411         return id;
00412     }
