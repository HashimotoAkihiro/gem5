00292     void clearLoadLocks(Request *req = NULL)
00293     {
00294         if (!req) {
00295             // No request, invaldate all locks to this line
00296             lockList.clear();
00297         } else {
00298             // Only invalidate locks that overlap with this request
00299             std::list<Lock>::iterator lock_itr = lockList.begin();
00300             while (lock_itr != lockList.end()) {
00301                 if (lock_itr->overlapping(req)) {
00302                     lock_itr = lockList.erase(lock_itr);
00303                 } else {
00304                     ++lock_itr;
00305                 }
00306             }
00307         }
00308     }

00351     bool checkWrite(PacketPtr pkt)
00352     {
00353         Request *req = pkt->req;
00354         if (pkt->isLLSC()) {
00355             // it's a store conditional... have to check for matching
00356             // load locked.
00357             bool success = false;
00358 
00359             for (std::list<Lock>::iterator i = lockList.begin();
00360                  i != lockList.end(); ++i)
00361             {
00362                 if (i->matchesContext(req)) {
00363                     // it's a store conditional, and as far as the memory
00364                     // system can tell, the requesting context's lock is
00365                     // still valid.
00366                     success = true;
00367                     break;
00368                 }
00369             }
00370 
00371             req->setExtraData(success ? 1 : 0);
00372             clearLoadLocks(req);
00373             return success;
00374         } else {
00375             // for *all* stores (conditional or otherwise) we have to
00376             // clear the list of load-locks as they're all invalid now.
00377             clearLoadLocks(req);
00378             return true;
00379         }
