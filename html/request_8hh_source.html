00385     void splitOnVaddr(Addr split_addr, RequestPtr &req1, RequestPtr &req2)
00386     {
00387         assert(privateFlags.isSet(VALID_VADDR));
00388         assert(privateFlags.noneSet(VALID_PADDR));

00390         req1 = new Request;

00392         req2 = new Request;
