00196     bool
00197     match(Addr va, uint16_t asn, uint8_t _vmid, bool hypLookUp,
00198           bool secure_lookup, bool ignore_asn, uint8_t target_el) const
00199     {
00200         bool match = false;
00201         Addr v = vpn << N;
00202 
00203         if (valid && va >= v && va <= v + size && (secure_lookup == !nstid) &&
00204             (hypLookUp == isHyp))
00205         {
00206             if (target_el == 2 || target_el == 3)
00207                 match = (el == target_el);
00208             else
00209                 match = (el == 0) || (el == 1);
00210             if (match && !ignore_asn) {
00211                 match = global || (asn == asid);
00212             }
00213             if (match && nstid) {
00214                 match = isHyp || (_vmid == vmid);
00215             }
00216         }
00217         return match;
00218     }

00226     void
00227     updateAttributes()
00228     {
00229         uint64_t mask;
00230         uint64_t newBits;
00231 
00232         // chec bit 11 to determine if its currently LPAE or VMSA format.
00233         if ( attributes & (1 << 11) ) {
00234             newBits = ((outerShareable ? 0x2 :
00235                       shareable         ? 0x3 : 0) << 7);
00236             mask = 0x180;
00237         } else {
00253             newBits = ((outerShareable ? 0:1) << 10) |
00254                       ((shareable ? 1:0) << 7) |
00255                       (innerAttrs << 4) |
00256                       (outerAttrs << 2);
00257                       // TODO: Supersection bit
00258             mask = 0x4FC;
00259         }
00260         // common bits
00261         newBits |= ns << 9;  // NS bit
00262         mask    |= 1  << 9;
00263         // add in the new bits
00264         attributes &= ~mask;
00265         attributes |= newBits;
00266     }
