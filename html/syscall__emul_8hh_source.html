00121   public:
00122 
00123     BaseBufferArg(Addr _addr, int _size) : addr(_addr), size(_size)
00124     {
00125         bufPtr = new uint8_t[size];
00126         // clear out buffer: in case we only partially populate this,
00127         // and then do a copyOut(), we want to make sure we don't
00128         // introduce any random junk into the simulated address space
00129         memset(bufPtr, 0, size);
00130     }
00131 
00132     virtual ~BaseBufferArg() { delete [] bufPtr; }
00133 
00134     //
00135     // copy data into simulator space (read from target memory)
00136     //
00137     virtual bool copyIn(SETranslatingPortProxy &memproxy)
00138     {
00139         memproxy.readBlob(addr, bufPtr, size);
00140         return true;    // no EFAULT detection for now
00141     }
00142 
00143     //
00144     // copy data out of simulator space (write to target memory)
00145     //
00146     virtual bool copyOut(SETranslatingPortProxy &memproxy)
00147     {
00148         memproxy.writeBlob(addr, bufPtr, size);
00149         return true;    // no EFAULT detection for now
00150     }
00151 
00152   protected:
00153     Addr addr;
00154     int size;
00155     uint8_t *bufPtr;

00363 template <class OS>
00364 SyscallReturn
00365 futexFunc(SyscallDesc *desc, int callnum, LiveProcess *process,
00366           ThreadContext *tc)
00367 {
00368     int index_uaddr = 0;
00369     int index_op = 1;
00370     int index_val = 2;
00371     int index_timeout = 3;
00372 
00373     uint64_t uaddr = process->getSyscallArg(tc, index_uaddr);
00374     int op = process->getSyscallArg(tc, index_op);
00375     int val = process->getSyscallArg(tc, index_val);
00376     uint64_t timeout = process->getSyscallArg(tc, index_timeout);
00377 
00378     std::map<uint64_t, std::list<ThreadContext *> * >
00379         &futex_map = tc->getSystemPtr()->futexMap;
00380 
00381     DPRINTF(SyscallVerbose, "In sys_futex: Address=%llx, op=%d, val=%d\n",
00382             uaddr, op, val);
00383 
00384     op &= ~OS::TGT_FUTEX_PRIVATE_FLAG;
00385 
00386     if (op == OS::TGT_FUTEX_WAIT) {
00387         if (timeout != 0) {
00388             warn("sys_futex: FUTEX_WAIT with non-null timeout unimplemented;"
00389                  "we'll wait indefinitely");
00390         }
00391 
00392         uint8_t *buf = new uint8_t[sizeof(int)];
00393         tc->getMemProxy().readBlob((Addr)uaddr, buf, (int)sizeof(int));
00394         int mem_val = *((int *)buf);
00395         delete buf;
00396 
00397         if(val != mem_val) {
00398             DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, read: %d, "
00399                                     "expected: %d\n", mem_val, val);
00400             return -OS::TGT_EWOULDBLOCK;
00401         }
00402 
00403         // Queue the thread context
00404         std::list<ThreadContext *> * tcWaitList;
00405         if (futex_map.count(uaddr)) {
00406             tcWaitList = futex_map.find(uaddr)->second;
00407         } else {
00408             tcWaitList = new std::list<ThreadContext *>();
00409             futex_map.insert(std::pair< uint64_t,
00410                             std::list<ThreadContext *> * >(uaddr, tcWaitList));
00411         }
00412         tcWaitList->push_back(tc);
00413         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAIT, suspending calling "
00414                                 "thread context\n");
00415         tc->suspend();
00416         return 0;
00417     } else if (op == OS::TGT_FUTEX_WAKE){
00418         int wokenUp = 0;
00419         std::list<ThreadContext *> * tcWaitList;
00420         if (futex_map.count(uaddr)) {
00421             tcWaitList = futex_map.find(uaddr)->second;
00422             while (tcWaitList->size() > 0 && wokenUp < val) {
00423                 tcWaitList->front()->activate();
00424                 tcWaitList->pop_front();
00425                 wokenUp++;
00426             }
00427             if(tcWaitList->empty()) {
00428                 futex_map.erase(uaddr);
00429                 delete tcWaitList;
00430             }
00431         }
00432         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, activated %d waiting "
00433                                 "thread contexts\n", wokenUp);
00434         return wokenUp;
00435     } else {
00436         warn("sys_futex: op %d is not implemented, just returning...", op);
00437         return 0;
00438     }
00439 
00440 }

00502 {

00524     tgt->st_gid = TheISA::htog(tgt->st_gid);

00621 template <class OS>
00622 static SyscallReturn
00623 openFunc(SyscallDesc *desc, int callnum, LiveProcess *process,
00624          ThreadContext *tc, int index)
00625 {
00626     std::string path;
00627 
00628     if (!tc->getMemProxy().tryReadString(path,
00629                 process->getSyscallArg(tc, index)))
00630         return -EFAULT;
00631 
00632     if (path == "/dev/sysdev0") {
00633         // This is a memory-mapped high-resolution timer device on Alpha.
00634         // We don't support it, so just punt.
00635         warn("Ignoring open(%s, ...)\n", path);
00636         return -ENOENT;
00637     }
00638 
00639     int tgtFlags = process->getSyscallArg(tc, index);
00640     int mode = process->getSyscallArg(tc, index);
00641     int hostFlags = 0;
00642 
00643     // translate open flags
00644     for (int i = 0; i < OS::NUM_OPEN_FLAGS; i++) {
00645         if (tgtFlags & OS::openFlagTable[i].tgtFlag) {
00646             tgtFlags &= ~OS::openFlagTable[i].tgtFlag;
00647             hostFlags |= OS::openFlagTable[i].hostFlag;
00648         }
00649     }
00650 
00651     // any target flags left?
00652     if (tgtFlags != 0)
00653         warn("Syscall: open: cannot decode flags 0x%x", tgtFlags);
00654 
00655 #ifdef __CYGWIN32__
00656     hostFlags |= O_BINARY;
00657 #endif
00658 
00659     // Adjust path for current working directory
00660     path = process->fullPath(path);
00661 
00662     DPRINTF(SyscallVerbose, "opening file %s\n", path.c_str());
00663 
00664     int fd;
00665     int local_errno;
00666     if (startswith(path, "/proc/") || startswith(path, "/system/") ||
00667         startswith(path, "/platform/") || startswith(path, "/sys/")) {
00668         // It's a proc/sys entry and requires special handling
00669         fd = OS::openSpecialFile(path, process, tc);
00670         local_errno = ENOENT;
00671      } else {
00672         // open the file
00673         fd = open(path.c_str(), hostFlags, mode);
00674         local_errno = errno;
00675      }
00676 
00677     if (fd == -1)
00678         return -local_errno;
00679 
00680     return process->alloc_fd(fd, path.c_str(), hostFlags, mode, false);
00681 }

00810 template <class OS>
00811 SyscallReturn
00812 mremapFunc(SyscallDesc *desc, int callnum, LiveProcess *process, ThreadContext *tc)
00813 {
00814     int index = 0;
00815     Addr start = process->getSyscallArg(tc, index);
00816     uint64_t old_length = process->getSyscallArg(tc, index);
00817     uint64_t new_length = process->getSyscallArg(tc, index);
00818     uint64_t flags = process->getSyscallArg(tc, index);
00819     uint64_t provided_address = 0;
00820     bool use_provided_address = flags & OS::TGT_MREMAP_FIXED;
00821 
00822     if (use_provided_address)
00823         provided_address = process->getSyscallArg(tc, index);
00824 
00825     if ((start % TheISA::VMPageSize != 0) ||
00826         (new_length % TheISA::VMPageSize != 0) ||
00827         (provided_address % TheISA::VMPageSize != 0)) {
00828         warn("mremap failing: arguments not page aligned");
00829         return -EINVAL;
00830     }
00831 
00832     if (new_length > old_length) {
00833         if ((start + old_length) == process->mmap_end &&
00834             (!use_provided_address || provided_address == start)) {
00835             uint64_t diff = new_length - old_length;
00836             process->allocateMem(process->mmap_end, diff);
00837             process->mmap_end += diff;
00838             return start;
00839         } else {
00840             if (!use_provided_address && !(flags & OS::TGT_MREMAP_MAYMOVE)) {
00841                 warn("can't remap here and MREMAP_MAYMOVE flag not set\n");
00842                 return -ENOMEM;
00843             } else {
00844                 uint64_t new_start = use_provided_address ?
00845                     provided_address : process->mmap_end;
00846                 process->pTable->remap(start, old_length, new_start);
00847                 warn("mremapping to new vaddr %08p-%08p, adding %d\n",
00848                      new_start, new_start + new_length,
00849                      new_length - old_length);
00850                 // add on the remaining unallocated pages
00851                 process->allocateMem(new_start + old_length,
00852                                      new_length - old_length,
00853                                      use_provided_address /* clobber */);
00854                 if (!use_provided_address)
00855                     process->mmap_end += new_length;
00856                 if (use_provided_address &&
00857                     new_start + new_length > process->mmap_end) {
00858                     // something fishy going on here, at least notify the user
00859                     // @todo: increase mmap_end?
00860                     warn("mmap region limit exceeded with MREMAP_FIXED\n");
00861                 }
00862                 warn("returning %08p as start\n", new_start);
00863                 return new_start;
00864             }
00865         }
00866     } else {
00867         if (use_provided_address && provided_address != start)
00868             process->pTable->remap(start, new_length, provided_address);
00869         process->pTable->unmap(start + new_length, old_length - new_length);
00870         return use_provided_address ? provided_address : start;
00871     }
00872 }

01198 template <class OS>
01199 SyscallReturn
01200 mmapFunc(SyscallDesc *desc, int num, LiveProcess *p, ThreadContext *tc)
01201 {
01202     int index = 0;
01203     Addr start = p->getSyscallArg(tc, index);
01204     uint64_t length = p->getSyscallArg(tc, index);
01205     index++; // int prot = p->getSyscallArg(tc, index);
01206     int flags = p->getSyscallArg(tc, index);
01207     int tgt_fd = p->getSyscallArg(tc, index);
01208     // int offset = p->getSyscallArg(tc, index);
01209 
01210     if (length > 0x100000000ULL)
01211         warn("mmap length argument %#x is unreasonably large.\n", length);
01212 
01213     if (!(flags & OS::TGT_MAP_ANONYMOUS)) {
01214         Process::FdMap *fd_map = p->sim_fd_obj(tgt_fd);
01215         if (!fd_map || fd_map->fd < 0) {
01216             warn("mmap failing: target fd %d is not valid\n", tgt_fd);
01217             return -EBADF;
01218         }
01219 
01220         if (fd_map->filename != "/dev/zero") {
01221             // This is very likely broken, but leave a warning here
01222             // (rather than panic) in case /dev/zero is known by
01223             // another name on some platform
01224             warn("allowing mmap of file %s; mmap not supported on files"
01225                  " other than /dev/zero\n", fd_map->filename);
01226         }
01227     }
01228 
01229     if ((start  % TheISA::VMPageSize) != 0 ||
01230         (length % TheISA::VMPageSize) != 0) {
01231         warn("mmap failing: arguments not page-aligned: "
01232              "start 0x%x length 0x%x",
01233              start, length);
01234         return -EINVAL;
01235     }
01236 
01237     // are we ok with clobbering existing mappings?  only set this to
01238     // true if the user has been warned.
01239     bool clobber = false;
01240 
01241     // try to use the caller-provided address if there is one
01242     bool use_provided_address = (start != 0);
01243 
01244     if (use_provided_address) {
01245         // check to see if the desired address is already in use
01246         if (!p->pTable->isUnmapped(start, length)) {
01247             // there are existing mappings in the desired range
01248             // whether we clobber them or not depends on whether the caller
01249             // specified MAP_FIXED
01250             if (flags & OS::TGT_MAP_FIXED) {
01251                 // MAP_FIXED specified: clobber existing mappings
01252                 warn("mmap: MAP_FIXED at 0x%x overwrites existing mappings\n",
01253                      start);
01254                 clobber = true;
01255             } else {
01256                 // MAP_FIXED not specified: ignore suggested start address
01257                 warn("mmap: ignoring suggested map address 0x%x\n", start);
01258                 use_provided_address = false;
01259             }
01260         }
01261     }
01262 
01263     if (!use_provided_address) {
01264         // no address provided, or provided address unusable:
01265         // pick next address from our "mmap region"
01266         if (OS::mmapGrowsDown()) {
01267             start = p->mmap_end - length;
01268             p->mmap_end = start;
01269         } else {
01270             start = p->mmap_end;
01271             p->mmap_end += length;
01272         }
01273     }
01274 
01275     p->allocateMem(start, length, clobber);
01276 
01277     return start;
01278 }
