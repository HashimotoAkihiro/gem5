00364         void
00365         advance()
00366         {
00367             Base::advance();
00368             flags = nextFlags;
00369             npc(pc() + (thumb() ? 2 : 4));
00370 
00371             if (_nextItstate) {
00372                 _itstate = _nextItstate;
00373                 _nextItstate = 0;
00374             } else if (_itstate) {
00375                 ITSTATE it = _itstate;
00376                 uint8_t cond_mask = it.mask;
00377                 uint8_t thumb_cond = it.cond;
00378                 DPRINTF(Decoder, "Advancing ITSTATE from %#x,%#x.\n",
00379                         thumb_cond, cond_mask);
00380                 cond_mask <<= 1;
00381                 uint8_t new_bit = bits(cond_mask, 4);
00382                 cond_mask &= mask(4);
00383                 if (cond_mask == 0)
00384                     thumb_cond = 0;
00385                 else
00386                     replaceBits(thumb_cond, 0, new_bit);
00387                 DPRINTF(Decoder, "Advancing ITSTATE to %#x,%#x.\n",
00388                         thumb_cond, cond_mask);
00389                 it.mask = cond_mask;
00390                 it.cond = thumb_cond;
00391                 _itstate = it;
00392             }
00393         }

00427         // Perform an interworking branch.
00428         void
00429         instIWNPC(Addr val)
00430         {
00431             bool thumbEE = (thumb() && jazelle());
00432 
00433             Addr newPC = val;
00434             if (thumbEE) {
00435                 if (bits(newPC, 0)) {
00436                     newPC = newPC & ~mask(1);
00437                 }  // else we have a bad interworking address; do not call
00438                    // panic() since the instruction could be executed
00439                    // speculatively
00440             } else {
00441                 if (bits(newPC, 0)) {
00442                     nextThumb(true);
00443                     newPC = newPC & ~mask(1);
00444                 } else if (!bits(newPC, 1)) {
00445                     nextThumb(false);
00446                 } else {
00447                     // This state is UNPREDICTABLE in the ARM architecture
00448                     // The easy thing to do is just mask off the bit and
00449                     // stay in the current mode, so we'll do that.
00450                     newPC &= ~mask(2);
00451                 }
00452             }
00453             npc(newPC);
00454         }
