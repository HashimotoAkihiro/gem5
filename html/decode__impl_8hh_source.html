00286 void
00287 DefaultDecode<Impl>::squash(DynInstPtr &inst, ThreadID tid)
00288 {
00289     DPRINTF(Decode, "[tid:%i]: [sn:%i] Squashing due to incorrect branch "
00290             "prediction detected at decode.\n", tid, inst->seqNum);
00291 
00292     // Send back mispredict information.
00293     toFetch->decodeInfo[tid].branchMispredict = true;
00294     toFetch->decodeInfo[tid].predIncorrect = true;
00295     toFetch->decodeInfo[tid].mispredictInst = inst;
00296     toFetch->decodeInfo[tid].squash = true;
00297     toFetch->decodeInfo[tid].doneSeqNum = inst->seqNum;
00298     toFetch->decodeInfo[tid].nextPC = inst->branchTarget();
00299     toFetch->decodeInfo[tid].branchTaken = inst->pcState().branching();
00300     toFetch->decodeInfo[tid].squashInst = inst;
00301     if (toFetch->decodeInfo[tid].mispredictInst->isUncondCtrl()) {
00302             toFetch->decodeInfo[tid].branchTaken = true;
00303     }
00304 
00305     InstSeqNum squash_seq_num = inst->seqNum;
00306 
00307     // Might have to tell fetch to unblock.
00308     if (decodeStatus[tid] == Blocked ||
00309         decodeStatus[tid] == Unblocking) {
00310         toFetch->decodeUnblock[tid] = 1;
00311     }
00312 
00313     // Set status to squashing.
00314     decodeStatus[tid] = Squashing;
00315 
00316     for (int i=0; i<fromFetch->size; i++) {
00317         if (fromFetch->insts[i]->threadNumber == tid &&
00318             fromFetch->insts[i]->seqNum > squash_seq_num) {
00319             fromFetch->insts[i]->setSquashed();
00320         }
00321     }
00322 
00323     // Clear the instruction list and skid buffer in case they have any
00324     // insts in them.
00325     while (!insts[tid].empty()) {
00326         insts[tid].pop();
00327     }
00328 
00329     while (!skidBuffer[tid].empty()) {
00330         skidBuffer[tid].pop();
00331     }
00332 
00333     // Squash instructions up until this one
00334     cpu->removeInstsUntil(squash_seq_num, tid);
00335 }

00337 template<class Impl>
00338 unsigned
00339 DefaultDecode<Impl>::squash(ThreadID tid)
00340 {
00341     DPRINTF(Decode, "[tid:%i]: Squashing.\n",tid);
00342 
00343     if (decodeStatus[tid] == Blocked ||
00344         decodeStatus[tid] == Unblocking) {
00345         if (FullSystem) {
00346             toFetch->decodeUnblock[tid] = 1;
00347         } else {
00348             // In syscall emulation, we can have both a block and a squash due
00349             // to a syscall in the same cycle.  This would cause both signals
00350             // to be high.  This shouldn't happen in full system.
00351             // @todo: Determine if this still happens.
00352             if (toFetch->decodeBlock[tid])
00353                 toFetch->decodeBlock[tid] = 0;
00354             else
00355                 toFetch->decodeUnblock[tid] = 1;
00356         }
00357     }
00358 
00359     // Set status to squashing.
00360     decodeStatus[tid] = Squashing;
00361 
00362     // Go through incoming instructions from fetch and squash them.
00363     unsigned squash_count = 0;
00364 
00365     for (int i=0; i<fromFetch->size; i++) {
00366         if (fromFetch->insts[i]->threadNumber == tid) {
00367             fromFetch->insts[i]->setSquashed();
00368             squash_count++;
00369         }
00370     }
00371 
00372     // Clear the instruction list and skid buffer in case they have any
00373     // insts in them.
00374     while (!insts[tid].empty()) {
00375         insts[tid].pop();
00376     }
00377 
00378     while (!skidBuffer[tid].empty()) {
00379         skidBuffer[tid].pop();
00380     }
00381 
00382     return squash_count;
00383 }

00506 template <class Impl>
00507 bool
00508 DefaultDecode<Impl>::checkSignalsAndUpdate(ThreadID tid)
00509 {
00510     // Check if there's a squash signal, squash if there is.
00511     // Check stall signals, block if necessary.
00512     // If status was blocked
00513     //     Check if stall conditions have passed
00514     //         if so then go to unblocking
00515     // If status was Squashing
00516     //     check if squashing is not high.  Switch to running this cycle.
00517 
00518     // Update the per thread stall statuses.
00519     readStallSignals(tid);
00520 
00521     // Check squash signals from commit.
00522     if (fromCommit->commitInfo[tid].squash) {
00523 
00524         DPRINTF(Decode, "[tid:%u]: Squashing instructions due to squash "
00525                 "from commit.\n", tid);
00526 
00527         squash(tid);
00528 
00529         return true;
00530     }
00531 
00532     // Check ROB squash signals from commit.
00533     if (fromCommit->commitInfo[tid].robSquashing) {
00534         DPRINTF(Decode, "[tid:%u]: ROB is still squashing.\n", tid);
00535 
00536         // Continue to squash.
00537         decodeStatus[tid] = Squashing;
00538 
00539         return true;
00540     }
00541 
00542     if (checkStall(tid)) {
00543         return block(tid);
00544     }
00545 
00546     if (decodeStatus[tid] == Blocked) {
00547         DPRINTF(Decode, "[tid:%u]: Done blocking, switching to unblocking.\n",
00548                 tid);
00549 
00550         decodeStatus[tid] = Unblocking;
00551 
00552         unblock(tid);
00553 
00554         return true;
00555     }
00556 
00557     if (decodeStatus[tid] == Squashing) {
00558         // Switch status to running if decode isn't being told to block or
00559         // squash this cycle.
00560         DPRINTF(Decode, "[tid:%u]: Done squashing, switching to running.\n",
00561                 tid);
00562 
00563         decodeStatus[tid] = Running;
00564 
00565         return false;
00566     }
00567 
00568     // If we've reached this point, we have not gotten any signals that
00569     // cause decode to change its status.  Decode remains the same as before.
00570     return false;
00571 }

00609 template<class Impl>
00610 void
00611 DefaultDecode<Impl>::decode(bool &status_change, ThreadID tid)
00612 {
00613     // If status is Running or idle,
00614     //     call decodeInsts()
00615     // If status is Unblocking,
00616     //     buffer any instructions coming from fetch
00617     //     continue trying to empty skid buffer
00618     //     check if stall conditions have passed
00619 
00620     if (decodeStatus[tid] == Blocked) {
00621         ++decodeBlockedCycles;
00622     } else if (decodeStatus[tid] == Squashing) {
00623         ++decodeSquashCycles;
00624     }
00625 
00626     // Decode should try to decode as many instructions as its bandwidth
00627     // will allow, as long as it is not currently blocked.
00628     if (decodeStatus[tid] == Running ||
00629         decodeStatus[tid] == Idle) {
00630         DPRINTF(Decode, "[tid:%u]: Not blocked, so attempting to run "
00631                 "stage.\n",tid);
00632 
00633         decodeInsts(tid);
00634     } else if (decodeStatus[tid] == Unblocking) {
00635         // Make sure that the skid buffer has something in it if the
00636         // status is unblocking.
00637         assert(!skidsEmpty());
00638 
00639         // If the status was unblocking, then instructions from the skid
00640         // buffer were used.  Remove those instructions and handle
00641         // the rest of unblocking.
00642         decodeInsts(tid);
00643 
00644         if (fetchInstsValid()) {
00645             // Add the current inputs to the skid buffer so they can be
00646             // reprocessed when this stage unblocks.
00647             skidInsert(tid);
00648         }
00649 
00650         status_change = unblock(tid) || status_change;
00651     }
00652 }

00654 template <class Impl>
00655 void
00656 DefaultDecode<Impl>::decodeInsts(ThreadID tid)
00657 {
00658     // Instructions can come either from the skid buffer or the list of
00659     // instructions coming from fetch, depending on decode's status.
00660     int insts_available = decodeStatus[tid] == Unblocking ?
00661         skidBuffer[tid].size() : insts[tid].size();
00662 
00663     if (insts_available == 0) {
00664         DPRINTF(Decode, "[tid:%u] Nothing to do, breaking out"
00665                 " early.\n",tid);
00666         // Should I change the status to idle?
00667         ++decodeIdleCycles;
00668         return;
00669     } else if (decodeStatus[tid] == Unblocking) {
00670         DPRINTF(Decode, "[tid:%u] Unblocking, removing insts from skid "
00671                 "buffer.\n",tid);
00672         ++decodeUnblockCycles;
00673     } else if (decodeStatus[tid] == Running) {
00674         ++decodeRunCycles;
00675     }
00676 
00677     DynInstPtr inst;
00678 
00679     std::queue<DynInstPtr>
00680         &insts_to_decode = decodeStatus[tid] == Unblocking ?
00681         skidBuffer[tid] : insts[tid];
00682 
00683     DPRINTF(Decode, "[tid:%u]: Sending instruction to rename.\n",tid);
00684 
00685     while (insts_available > 0 && toRenameIndex < decodeWidth) {
00686         assert(!insts_to_decode.empty());
00687 
00688         inst = insts_to_decode.front();
00689 
00690         insts_to_decode.pop();
