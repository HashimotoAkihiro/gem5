00615 template<class Impl>
00616 void
00617 DefaultIEW<Impl>::instToCommit(DynInstPtr &inst)
00618 {
00619     // This function should not be called after writebackInsts in a
00620     // single cycle.  That will cause problems with an instruction
00621     // being added to the queue to commit without being processed by
00622     // writebackInsts prior to being sent to commit.
00623 
00624     // First check the time slot that this instruction will write
00625     // to.  If there are free write ports at the time, then go ahead
00626     // and write the instruction to that time.  If there are not,
00627     // keep looking back to see where's the first time there's a
00628     // free slot.
00629     while ((*iewQueue)[wbCycle].insts[wbNumInst]) {
00630         ++wbNumInst;
00631         if (wbNumInst == wbWidth) {
00632             ++wbCycle;
00633             wbNumInst = 0;
00634         }
00635 
00636         assert((wbCycle * wbWidth + wbNumInst) <= wbMax);
00637     }
00638 
00639     DPRINTF(IEW, "Current wb cycle: %i, width: %i, numInst: %i\nwbActual:%i\n",
00640             wbCycle, wbWidth, wbNumInst, wbCycle * wbWidth + wbNumInst);
00641     // Add finished instruction to queue to commit.
00642     (*iewQueue)[wbCycle].insts[wbNumInst] = inst;
00643     (*iewQueue)[wbCycle].size++;
00644 }

00818 void
00819 DefaultIEW<Impl>::checkSignalsAndUpdate(ThreadID tid)
00820 {
00821     // Check if there's a squash signal, squash if there is
00822     // Check stall signals, block if there is.
00823     // If status was Blocked
00824     //     if so then go to unblocking
00825     // If status was Squashing
00826     //     check if squashing is not high.  Switch to running this cycle.
00827 
00828     readStallSignals(tid);
00829 
00830     if (fromCommit->commitInfo[tid].squash) {
00831         squash(tid);
00832 
00833         if (dispatchStatus[tid] == Blocked ||
00834             dispatchStatus[tid] == Unblocking) {
00835             toRename->iewUnblock[tid] = true;
00836             wroteToTimeBuffer = true;
00837         }
00838 
00839         dispatchStatus[tid] = Squashing;
00840         fetchRedirect[tid] = false;
00841         return;
00842     }
00843 
00844     if (fromCommit->commitInfo[tid].robSquashing) {
00845         DPRINTF(IEW, "[tid:%i]: ROB is still squashing.\n", tid);
00846 
00847         dispatchStatus[tid] = Squashing;
00848         emptyRenameInsts(tid);
00849         wroteToTimeBuffer = true;
00850         return;
00851     }
00852 
00853     if (checkStall(tid)) {
00854         block(tid);
00855         dispatchStatus[tid] = Blocked;
00856         return;
00857     }
00858 
00859     if (dispatchStatus[tid] == Blocked) {
00860         // Status from previous cycle was blocked, but there are no more stall
00861         // conditions.  Switch over to unblocking.
00862         DPRINTF(IEW, "[tid:%i]: Done blocking, switching to unblocking.\n",
00863                 tid);
00864 
00865         dispatchStatus[tid] = Unblocking;
00866 
00867         unblock(tid);
00868 
00869         return;
00870     }
00871 
00872     if (dispatchStatus[tid] == Squashing) {
00873         // Switch status to running if rename isn't being told to block or
00874         // squash this cycle.
00875         DPRINTF(IEW, "[tid:%i]: Done squashing, switching to running.\n",
00876                 tid);
00877 
00878         dispatchStatus[tid] = Running;
00879 
00880         return;
00881     }

00948 template<class Impl>
00949 void
00950 DefaultIEW<Impl>::dispatch(ThreadID tid)
00951 {
00952     // If status is Running or idle,
00953     //     call dispatchInsts()
00954     // If status is Unblocking,
00955     //     buffer any instructions coming from rename
00956     //     continue trying to empty skid buffer
00957     //     check if stall conditions have passed
00958 
00959     if (dispatchStatus[tid] == Blocked) {
00960         ++iewBlockCycles;
00961 
00962     } else if (dispatchStatus[tid] == Squashing) {
00963         ++iewSquashCycles;
00964     }
00965 
00966     // Dispatch should try to dispatch as many instructions as its bandwidth
00967     // will allow, as long as it is not currently blocked.
00968     if (dispatchStatus[tid] == Running ||
00969         dispatchStatus[tid] == Idle) {
00970         DPRINTF(IEW, "[tid:%i] Not blocked, so attempting to run "
00971                 "dispatch.\n", tid);
00972 
00973         dispatchInsts(tid);
00974     } else if (dispatchStatus[tid] == Unblocking) {
00975         // Make sure that the skid buffer has something in it if the
00976         // status is unblocking.
00977         assert(!skidsEmpty());
00978 
00979         // If the status was unblocking, then instructions from the skid
00980         // buffer were used.  Remove those instructions and handle
00981         // the rest of unblocking.
00982         dispatchInsts(tid);
00983 
00984         ++iewUnblockCycles;
00985 
00986         if (validInstsFromRename()) {
00987             // Add the current inputs to the skid buffer so they can be
00988             // reprocessed when this stage unblocks.
00989             skidInsert(tid);
00990         }
00991 
00992         unblock(tid);
00993     }
00994 }

00996 template <class Impl>
00997 void
00998 DefaultIEW<Impl>::dispatchInsts(ThreadID tid)
00999 {
01000     // Obtain instructions from skid buffer if unblocking, or queue from rename
01001     // otherwise.
01002     std::queue<DynInstPtr> &insts_to_dispatch =
01003         dispatchStatus[tid] == Unblocking ?
01004         skidBuffer[tid] : insts[tid];
01005 
01006     int insts_to_add = insts_to_dispatch.size();
01007 
01008     DynInstPtr inst;
01009     bool add_to_iq = false;
01010     int dis_num_inst = 0;
01011 
01012     // Loop through the instructions, putting them in the instruction
01013     // queue.
01014     for ( ; dis_num_inst < insts_to_add &&
01015               dis_num_inst < dispatchWidth;
01016           ++dis_num_inst)
01017     {
01018         inst = insts_to_dispatch.front();
01019 
01020         if (dispatchStatus[tid] == Unblocking) {
01021             DPRINTF(IEW, "[tid:%i]: Issue: Examining instruction from skid "
01022                     "buffer\n", tid);
01023         }
01024 
01025         // Make sure there's a valid instruction there.
01026         assert(inst);
01027 
01028         DPRINTF(IEW, "[tid:%i]: Issue: Adding PC %s [sn:%lli] [tid:%i] to "
01029                 "IQ.\n",
01030                 tid, inst->pcState(), inst->seqNum, inst->threadNumber);
01031 
01032         // Be sure to mark these instructions as ready so that the
01033         // commit stage can go ahead and execute them, and mark
01034         // them as issued so the IQ doesn't reprocess them.
01035 
01036         // Check for squashed instructions.
01037         if (inst->isSquashed()) {
01038             DPRINTF(IEW, "[tid:%i]: Issue: Squashed instruction encountered, "
01039                     "not adding to IQ.\n", tid);
01040 
01041             ++iewDispSquashedInsts;
01042 
01043             insts_to_dispatch.pop();
01044 
01045             //Tell Rename That An Instruction has been processed
01046             if (inst->isLoad() || inst->isStore()) {
01047                 toRename->iewInfo[tid].dispatchedToLSQ++;
01048             }
01049             toRename->iewInfo[tid].dispatched++;
01050 
01051             continue;
01052         }
01053 
01054         // Check for full conditions.
01055         if (instQueue.isFull(tid)) {
01056             DPRINTF(IEW, "[tid:%i]: Issue: IQ has become full.\n", tid);
01057 
01058             // Call function to start blocking.
01059             block(tid);
01060 
01061             // Set unblock to false. Special case where we are using
01062             // skidbuffer (unblocking) instructions but then we still
01063             // get full in the IQ.
01064             toRename->iewUnblock[tid] = false;
01065 
01066             ++iewIQFullEvents;
01067             break;
01068         } else if (ldstQueue.isFull(tid)) {
01069             DPRINTF(IEW, "[tid:%i]: Issue: LSQ has become full.\n",tid);
01070 
01071             // Call function to start blocking.
01072             block(tid);
01073 
01074             // Set unblock to false. Special case where we are using
01075             // skidbuffer (unblocking) instructions but then we still
01076             // get full in the IQ.
01077             toRename->iewUnblock[tid] = false;
01078 
01079             ++iewLSQFullEvents;
01080             break;
01081         }
01082 
01083         // Otherwise issue the instruction just fine.
01084         if (inst->isLoad()) {
01085             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01086                     "encountered, adding to LSQ.\n", tid);
01087 
01088             // Reserve a spot in the load store queue for this
01089             // memory access.
01090             ldstQueue.insertLoad(inst);
01091 
01092             ++iewDispLoadInsts;
01093 
01094             add_to_iq = true;
01095 
01096             toRename->iewInfo[tid].dispatchedToLSQ++;
01097         } else if (inst->isStore()) {
01098             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01099                     "encountered, adding to LSQ.\n", tid);
01100 
01101             ldstQueue.insertStore(inst);
01102 
01103             ++iewDispStoreInsts;
01104 
01105             if (inst->isStoreConditional()) {
01106                 // Store conditionals need to be set as "canCommit()"
01107                 // so that commit can process them when they reach the
01108                 // head of commit.
01109                 // @todo: This is somewhat specific to Alpha.
01110                 inst->setCanCommit();
01111                 instQueue.insertNonSpec(inst);
01112                 add_to_iq = false;
01113 
01114                 ++iewDispNonSpecInsts;
01115             } else {
01116                 add_to_iq = true;
01117             }
01118 
01119             toRename->iewInfo[tid].dispatchedToLSQ++;
01120         } else if (inst->isMemBarrier() || inst->isWriteBarrier()) {
01121             // Same as non-speculative stores.
01122             inst->setCanCommit();
01123             instQueue.insertBarrier(inst);
01124             add_to_iq = false;
01125         } else if (inst->isNop()) {
01126             DPRINTF(IEW, "[tid:%i]: Issue: Nop instruction encountered, "
01127                     "skipping.\n", tid);
01128 
01129             inst->setIssued();
01130             inst->setExecuted();
01131             inst->setCanCommit();
01132 
01133             instQueue.recordProducer(inst);
01134 
01135             iewExecutedNop[tid]++;
01136 
01137             add_to_iq = false;
01138         } else if (inst->isExecuted()) {
01139             assert(0 && "Instruction shouldn't be executed.\n");
01140             DPRINTF(IEW, "Issue: Executed branch encountered, "
01141                     "skipping.\n");
01142 
01143             inst->setIssued();
01144             inst->setCanCommit();
01145 
01146             instQueue.recordProducer(inst);
01147 
01148             add_to_iq = false;
01149         } else {
01150             add_to_iq = true;
01151         }
01152         if (inst->isNonSpeculative()) {
01153             DPRINTF(IEW, "[tid:%i]: Issue: Nonspeculative instruction "
01154                     "encountered, skipping.\n", tid);
01155 
01156             // Same as non-speculative stores.
01157             inst->setCanCommit();
01158 
01159             // Specifically insert it as nonspeculative.
01160             instQueue.insertNonSpec(inst);
01161 
01162             ++iewDispNonSpecInsts;
01163 
01164             add_to_iq = false;
01165         }
01166 
01167         // If the instruction queue is not full, then add the
01168         // instruction.
01169         if (add_to_iq) {
01170             instQueue.insert(inst);
01171         }
01172 
01173         insts_to_dispatch.pop();
01174 
01175         toRename->iewInfo[tid].dispatched++;
01176 
01177         ++iewDispatchedInsts;
01178 
01179 #if TRACING_ON
01180         inst->dispatchTick = curTick() - inst->fetchTick;
01181 #endif
01182         ppDispatch->notify(inst);
01183     }
01184 
01185     if (!insts_to_dispatch.empty()) {
01186         DPRINTF(IEW,"[tid:%i]: Issue: Bandwidth Full. Blocking.\n", tid);
01187         block(tid);
01188         toRename->iewUnblock[tid] = false;
01189     }
01190 
01191     if (dispatchStatus[tid] == Idle && dis_num_inst) {
01192         dispatchStatus[tid] = Running;
01193 
01194         updatedQueues = true;
01195     }
01196 
01197     dis_num_inst = 0;
01198 }

01223 template <class Impl>
01224 void
01225 DefaultIEW<Impl>::executeInsts()
01226 {
01227     wbNumInst = 0;
01228     wbCycle = 0;
01229 
01230     list<ThreadID>::iterator threads = activeThreads->begin();
01231     list<ThreadID>::iterator end = activeThreads->end();
01232 
01233     while (threads != end) {
01234         ThreadID tid = *threads++;
01235         fetchRedirect[tid] = false;
01236     }
01237 
01238     // Uncomment this if you want to see all available instructions.
01239     // @todo This doesn't actually work anymore, we should fix it.
01240 //    printAvailableInsts();
01241 
01242     // Execute/writeback any instructions that are available.
01243     int insts_to_execute = fromIssue->size;
01244     int inst_num = 0;
01245     for (; inst_num < insts_to_execute;
01246           ++inst_num) {
01247 
01248         DPRINTF(IEW, "Execute: Executing instructions from IQ.\n");
01249 
01250         DynInstPtr inst = instQueue.getInstToExecute();
01251 
01252         DPRINTF(IEW, "Execute: Processing PC %s, [tid:%i] [sn:%i].\n",
01253                 inst->pcState(), inst->threadNumber,inst->seqNum);
01254 
01255         // Check if the instruction is squashed; if so then skip it
01256         if (inst->isSquashed()) {
01257             DPRINTF(IEW, "Execute: Instruction was squashed. PC: %s, [tid:%i]"
01258                          " [sn:%i]\n", inst->pcState(), inst->threadNumber,
01259                          inst->seqNum);
01260 
01261             // Consider this instruction executed so that commit can go
01262             // ahead and retire the instruction.
01263             inst->setExecuted();
01264 
01265             // Not sure if I should set this here or just let commit try to
01266             // commit any squashed instructions.  I like the latter a bit more.
01267             inst->setCanCommit();
01268 
01269             ++iewExecSquashedInsts;
01270 
01271             decrWb(inst->seqNum);
01272             continue;
01273         }
01274 
01275         Fault fault = NoFault;
01276 
01277         // Execute instruction.
01278         // Note that if the instruction faults, it will be handled
01279         // at the commit stage.
01280         if (inst->isMemRef()) {
01281             DPRINTF(IEW, "Execute: Calculating address for memory "
01282                     "reference.\n");
01283 
01284             // Tell the LDSTQ to execute this instruction (if it is a load).
01285             if (inst->isLoad()) {
01286                 // Loads will mark themselves as executed, and their writeback
01287                 // event adds the instruction to the queue to commit
01288                 fault = ldstQueue.executeLoad(inst);
01289 
01290                 if (inst->isTranslationDelayed() &&
01291                     fault == NoFault) {
01292                     // A hw page table walk is currently going on; the
01293                     // instruction must be deferred.
01294                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01295                             "load.\n");
01296                     instQueue.deferMemInst(inst);
01297                     continue;
01298                 }
01299 
01300                 if (inst->isDataPrefetch() || inst->isInstPrefetch()) {
01301                     inst->fault = NoFault;
01302                 }
01303             } else if (inst->isStore()) {
01304                 fault = ldstQueue.executeStore(inst);
01305 
01306                 if (inst->isTranslationDelayed() &&
01307                     fault == NoFault) {
01308                     // A hw page table walk is currently going on; the
01309                     // instruction must be deferred.
01310                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01311                             "store.\n");
01312                     instQueue.deferMemInst(inst);
01313                     continue;
01314                 }
01315 
01316                 // If the store had a fault then it may not have a mem req
01317                 if (fault != NoFault || !inst->readPredicate() ||
01318                         !inst->isStoreConditional()) {
01319                     // If the instruction faulted, then we need to send it along
01320                     // to commit without the instruction completing.
01321                     // Send this instruction to commit, also make sure iew stage
01322                     // realizes there is activity.
01323                     inst->setExecuted();
01324                     instToCommit(inst);
01325                     activityThisCycle();
01326                 }
01327 
01328                 // Store conditionals will mark themselves as
01329                 // executed, and their writeback event will add the
01330                 // instruction to the queue to commit.
01331             } else {
01332                 panic("Unexpected memory type!\n");
01333             }
01334 
01335         } else {
01336             // If the instruction has already faulted, then skip executing it.
01337             // Such case can happen when it faulted during ITLB translation.
01338             // If we execute the instruction (even if it's a nop) the fault
01339             // will be replaced and we will lose it.
01340             if (inst->getFault() == NoFault) {
01341                 inst->execute();
01342                 if (!inst->readPredicate())
01343                     inst->forwardOldRegs();
01344             }
01345 
01346             inst->setExecuted();
01347 
01348             instToCommit(inst);
01349         }
01350 
01351         updateExeInstStats(inst);
01352 
01353         // Check if branch prediction was correct, if not then we need
01354         // to tell commit to squash in flight instructions.  Only
01355         // handle this if there hasn't already been something that
01356         // redirects fetch in this group of instructions.
01357 
01358         // This probably needs to prioritize the redirects if a different
01359         // scheduler is used.  Currently the scheduler schedules the oldest
01360         // instruction first, so the branch resolution order will be correct.
01361         ThreadID tid = inst->threadNumber;
01362 
01363         if (!fetchRedirect[tid] ||
01364             !toCommit->squash[tid] ||
01365             toCommit->squashedSeqNum[tid] > inst->seqNum) {
01366 
01367             // Prevent testing for misprediction on load instructions,
01368             // that have not been executed.
01369             bool loadNotExecuted = !inst->isExecuted() && inst->isLoad();
01370 
01371             if (inst->mispredicted() && !loadNotExecuted) {
01372                 fetchRedirect[tid] = true;
01373
