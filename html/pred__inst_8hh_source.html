00057 static inline uint32_t
00058 modified_imm(uint8_t ctrlImm, uint8_t dataImm)
00059 {
00060     uint32_t bigData = dataImm;
00061     uint32_t bigCtrl = ctrlImm;
00062     if (bigCtrl < 4) {
00063         switch (bigCtrl) {
00064           case 0:
00065             return bigData;
00066           case 1:
00067             return bigData | (bigData << 16);
00068           case 2:
00069             return (bigData << 8) | (bigData << 24);
00070           case 3:
00071             return (bigData << 0) | (bigData << 8) |
00072                    (bigData << 16) | (bigData << 24);
00073         }
00074     }
00075     bigCtrl = (bigCtrl << 1) | ((bigData >> 7) & 0x1);
00076     bigData |= (1 << 7);
00077     return bigData << (32 - bigCtrl);
00078 }

00080 static inline uint64_t
00081 simd_modified_imm(bool op, uint8_t cmode, uint8_t data, bool &immValid,
00082                   bool isAarch64 = false)
00083 {
00084     uint64_t bigData = data;
00085     immValid = true;
00086     switch (cmode) {
00087       case 0x0:
00088       case 0x1:
00089         bigData = (bigData << 0) | (bigData << 32);
00090         break;
00091       case 0x2:
00092       case 0x3:
00093         bigData = (bigData << 8) | (bigData << 40);
00094         break;
00095       case 0x4:
00096       case 0x5:
00097         bigData = (bigData << 16) | (bigData << 48);
00098         break;
00099       case 0x6:
00100       case 0x7:
00101         bigData = (bigData << 24) | (bigData << 56);
00102         break;
00103       case 0x8:
00104       case 0x9:
00105         bigData = (bigData << 0) | (bigData << 16) |
00106                   (bigData << 32) | (bigData << 48);
00107         break;
00108       case 0xa:
00109       case 0xb:
00110         bigData = (bigData << 8) | (bigData << 24) |
00111                   (bigData << 40) | (bigData << 56);
00112         break;
00113       case 0xc:
00114         bigData = (0xffULL << 0) | (bigData << 8) |
00115                   (0xffULL << 32) | (bigData << 40);
00116         break;
00117       case 0xd:
00118         bigData = (0xffffULL << 0) | (bigData << 16) |
00119                   (0xffffULL << 32) | (bigData << 48);
00120         break;
00121       case 0xe:
00122         if (op) {
00123             bigData = 0;
00124             for (int i = 7; i >= 0; i--) {
00125                 if (bits(data, i)) {
00126                     bigData |= (ULL(0xFF) << (i * 8));
00127                 }
00128             }
00129         } else {
00130             bigData = (bigData << 0)  | (bigData << 8)  |
00131                       (bigData << 16) | (bigData << 24) |
00132                       (bigData << 32) | (bigData << 40) |
00133                       (bigData << 48) | (bigData << 56);
00134         }
00135         break;
00136       case 0xf:
00137         {
00138             uint64_t bVal = 0;
00139             if (!op) {
00140                 bVal = bits(bigData, 6) ? (0x1F) : (0x20);
00141                 bigData = (bits(bigData, 5, 0) << 19) |
00142                           (bVal << 25) | (bits(bigData, 7) << 31);
00143                 bigData |= (bigData << 32);
00144                 break;
00145             } else if (isAarch64) {
00146                 bVal = bits(bigData, 6) ? (0x0FF) : (0x100);
00147                 bigData = (bits(bigData, 5, 0) << 48) |
00148                           (bVal << 54) | (bits(bigData, 7) << 63);
00149                 break;
00150             }
00151         }
00152         // Fall through, immediate encoding is invalid.
00153       default:
00154         immValid = false;
00155         break;
00156     }
00157     return bigData;
00158 }

00160 static inline uint64_t
00161 vfp_modified_imm(uint8_t data, bool wide)
00162 {
00163     uint64_t bigData = data;
00164     uint64_t repData;
00165     if (wide) {
00166         repData = bits(data, 6) ? 0xFF : 0;
00167         bigData = (bits(bigData, 5, 0) << 48) |
00168                   (repData << 54) | (bits(~bigData, 6) << 62) |
00169                   (bits(bigData, 7) << 63);
00170     } else {
00171         repData = bits(data, 6) ? 0x1F : 0;
00172         bigData = (bits(bigData, 5, 0) << 19) |
00173                   (repData << 25) | (bits(~bigData, 6) << 30) |
00174                   (bits(bigData, 7) << 31);
00175     }
00176     return bigData;
00177 }
