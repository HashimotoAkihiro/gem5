00172 // This function lets the compiler figure out how to call the
00173 // swap_byte functions above for different data types.  Since the
00174 // sizeof() values are known at compile time, it should inline to a
00175 // direct call to the right swap_byteNN() function.
00176 template <typename T>
00177 inline T swap_byte(T x) {
00178   //@@change sizeof(T) == 16 
00179   //  if (sizeof(T) == 16)
00180   //      return swap_byte128(x);
00181     if (sizeof(T) == 8)
00182         return swap_byte64((uint64_t)x);
00183     else if (sizeof(T) == 4)
00184         return swap_byte32((uint32_t)x);
00185     else if (sizeof(T) == 2)
00186         return swap_byte16((uint16_t)x);
00187     else if (sizeof(T) == 1)
00188         return x;
00189     else
00190         panic("Can't byte-swap values larger than 64 bits");
00191 }
