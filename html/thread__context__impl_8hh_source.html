00088 {
00089     DPRINTF(O3CPU, "Calling activate on Thread Context %d\n",
00090             threadId());
00091 
00092     if (thread->status() == ThreadContext::Active)
00093         return;
00094

00096     thread->setStatus(ThreadContext::Active);

00098     // status() == Suspended
