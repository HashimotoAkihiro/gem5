00508     void setBlocked(BlockedCause cause)
00509     {
00510         uint8_t flag = 1 << cause;
00511         if (blocked == 0) {
00512             blocked_causes[cause]++;
00513             blockedCycle = curCycle();
00514             cpuSidePort->setBlocked();
00515         }
00516         blocked |= flag;
00517         DPRINTF(Cache,"Blocking for cause %d, mask=%d\n", cause, blocked);
00518     }

00527     void clearBlocked(BlockedCause cause)
00528     {
00529         uint8_t flag = 1 << cause;
00530         blocked &= ~flag;
00531         DPRINTF(Cache,"Unblocking for cause %d, mask=%d\n", cause, blocked);
00532         if (blocked == 0) {
00533             blocked_cycles[cause] += curCycle() - blockedCycle;
00534             cpuSidePort->clearBlocked();
