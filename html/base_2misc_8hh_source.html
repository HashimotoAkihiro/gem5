00079 //
00080 // This implements a cprintf based panic() function.  panic() should
00081 // be called when something happens that should never ever happen
00082 // regardless of what the user does (i.e., an acutal m5 bug).  panic()
00083 // calls abort which can dump core or enter the debugger.
00084 //
00085 //
00086 #define panic(...) exit_message("panic", -1, __VA_ARGS__)

00088 //
00089 // This implements a cprintf based fatal() function.  fatal() should
00090 // be called when the simulation cannot continue due to some condition
00091 // that is the user's fault (bad configuration, invalid arguments,
00092 // etc.) and not a simulator bug.  fatal() calls  abort() like
00093 // panic() does.
00094 //
00095 #define fatal(...) exit_message("fatal", -1, __VA_ARGS__)

00146 // Only print the message the first time this expression is
00147 // encountered.  i.e.  This doesn't check the string itself and
00148 // prevent duplicate strings, this prevents the statement from
00149 // happening more than once. So, even if the arguments change and that
00150 // would have resulted in a different message thoes messages would be
00151 // supressed.
00152 #define base_message_once(...) do {                     \
00153         static bool once = false;                       \
00154         if (!once) {                                    \
00155             base_message(__VA_ARGS__);                  \
00156             once = true;                                \
00157         }                                               \
00158     } while (0)
