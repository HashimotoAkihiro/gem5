00309 template <class Impl>
00310 void
00311 FrontEnd<Impl>::tick()
00312 {
00313     if (switchedOut)
00314         return;
00315 
00316     for (int insts_to_queue = numInstsReady[-latency];
00317          !instBuffer.empty() && insts_to_queue;
00318          --insts_to_queue)
00319     {
00320         DPRINTF(FE, "Transferring instruction [sn:%lli] to the feBuffer\n",
00321                 instBuffer.front()->seqNum);
00322         feBuffer.push_back(instBuffer.front());
00323         instBuffer.pop_front();
00324     }
00325 
00326     numInstsReady.advance();
00327 
00328     // @todo: Maybe I want to just have direct communication...
00329     if (fromCommit->doneSeqNum) {
00330         branchPred.update(fromCommit->doneSeqNum, 0);
00331     }
00332 
00333     IFQCount += instBufferSize;
00334     IFQFcount += instBufferSize == maxInstBufferSize;
00335 
00336     // Fetch cache line
00337     if (status == IcacheAccessComplete) {
00338         cacheBlkValid = true;
00339 
00340         status = Running;
00341 //        if (barrierInst)
00342 //            status = SerializeBlocked;
00343         if (freeRegs <= 0)
00344             status = RenameBlocked;
00345         checkBE();
00346     } else if (status == IcacheWaitResponse || status == IcacheWaitRetry) {
00347         DPRINTF(FE, "Still in Icache wait.\n");
00348         icacheStallCycles++;
00349         return;
00350     }
00351 
00352     if (status == RenameBlocked || status == SerializeBlocked ||
00353         status == TrapPending || status == BEBlocked) {
00354         // Will cause a one cycle bubble between changing state and
00355         // restarting.
00356         DPRINTF(FE, "In blocked status.\n");
00357 
00358         fetchBlockedCycles++;
00359 
00360         if (status == SerializeBlocked) {
00361             dispatchSerializeStallCycles++;
00362         }
00363         updateStatus();
00364         return;
00365     } else if (status == QuiescePending) {
00366         DPRINTF(FE, "Waiting for quiesce to execute or get squashed.\n");
00367         return;
00368     } else if (status != IcacheAccessComplete) {
00369         if (fetchCacheLineNextCycle) {
00370             Fault fault = fetchCacheLine();
00371             if (fault != NoFault) {
00372                 handleFault(fault);
00373                 fetchFault = fault;
00374                 return;
00375             }
00376             fetchCacheLineNextCycle = false;
00377         }
00378         // If miss, stall until it returns.
00379         if (status == IcacheWaitResponse || status == IcacheWaitRetry) {
00380             // Tell CPU to not tick me for now.
00381             return;
00382         }
00383     }
00384 
00385     fetchCycles++;
00386 
00387     int num_inst = 0;
00388 
00389     // Otherwise loop and process instructions.
00390     // One way to hack infinite width is to set width and maxInstBufferSize
00391     // both really high.  Inelegant, but probably will work.
00392     while (num_inst < width &&
00393            instBufferSize < maxInstBufferSize) {
00394         // Get instruction from cache line.
00395         DynInstPtr inst = getInstFromCacheline();
00396 
00397         if (!inst) {
00398             // PC is no longer in the cache line, end fetch.
00399             // Might want to check this at the end of the cycle so that
00400             // there's no cycle lost to checking for a new cache line.
00401             DPRINTF(FE, "Need to get new cache line\n");
00402             fetchCacheLineNextCycle = true;
00403             break;
00404         }
00405 
00406         processInst(inst);
00407 
00408         if (status == SerializeBlocked) {
00409             break;
00410         }
00411 
00412         // Possibly push into a time buffer that estimates the front end
00413         // latency
00414         instBuffer.push_back(inst);
00415         ++instBufferSize;
00416         numInstsReady[0]++;
00417         ++num_inst;
00418 
00419         if (inst->isQuiesce()) {
00420             status = QuiescePending;
00421             break;
00422         }
00423 
00424         if (inst->predTaken()) {
00425             // Start over with tick?
00426             break;
00427         } else if (freeRegs <= 0) {
00428             DPRINTF(FE, "Ran out of free registers to rename to!\n");
00429             status = RenameBlocked;
00430             break;
00431         } else if (serializeNext) {
00432             break;
00433         }
00434     }
00435 
00436     fetchNisnDist.sample(num_inst);
00437     checkBE();
00438 
00439     DPRINTF(FE, "Num insts processed: %i, Inst Buffer size: %i, Free "
00440             "Regs %i\n", num_inst, instBufferSize, freeRegs);
00441 }
