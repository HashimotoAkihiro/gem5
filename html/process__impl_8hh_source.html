00042 template<class AddrType>
00043 void
00044 copyStringArray(std::vector<std::string> &strings,
00045         AddrType array_ptr, AddrType data_ptr,
00046         SETranslatingPortProxy& memProxy)
00047 {
00048     AddrType data_ptr_swap;
00049     for (std::vector<std::string>::size_type i = 0; i < strings.size(); ++i) {
00050         data_ptr_swap = TheISA::htog(data_ptr);
00051         memProxy.writeBlob(array_ptr, (uint8_t*)&data_ptr_swap,
00052                 sizeof(AddrType));
00053         memProxy.writeString(data_ptr, strings[i].c_str());
00054         array_ptr += sizeof(AddrType);
00055         data_ptr += strings[i].size() + 1;
00056     }
00057     // add NULL terminator
00058     data_ptr = 0;
00059 
00060     memProxy.writeBlob(array_ptr, (uint8_t*)&data_ptr, sizeof(AddrType));
00061 }
