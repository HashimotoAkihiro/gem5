00145     inline bool
00146     ctrOk(uint32_t& ctr) const
00147     {
00148         bool ctr_ok;
00149         if (bo & 4) {
00150             ctr_ok = true;
00151         } else {
00152             ctr--;
00153             if (ctr != 0) {
00154                 ctr_ok = ((bo & 2) == 0);
00155             } else {
00156                 ctr_ok = ((bo & 2) != 0);
00157             }
00158         }
00159         return ctr_ok;
00160     }
00161 
00162     inline bool
00163     condOk(uint32_t cr) const
00164     {
00165         bool cond_ok;
00166         if (bo & 16) {
00167             cond_ok = true;
00168         } else {
00169             cond_ok = (((cr >> (31 - bi)) & 1) == ((bo >> 3) & 1));
00170         }
00171         return cond_ok;
