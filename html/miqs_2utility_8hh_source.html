00074 static inline bool
00075 inUserMode(ThreadContext *tc)
00076 {
00077     MiscReg Stat = tc->readMiscReg(MISCREG_STATUS);
00078     MiscReg Dbg = tc->readMiscReg(MISCREG_DEBUG);
00079 
00080     if ((Stat & 0x10000006) == 0 &&  // EXL, ERL or CU0 set, CP0 accessible
00081         (Dbg & 0x40000000) == 0 &&   // DM bit set, CP0 accessible
00082         (Stat & 0x00000018) != 0) {  // KSU = 0, kernel mode is base mode
00083         // Unable to use Status_CU0, etc directly, using bitfields & masks
00084         return true;
00085     } else {
00086         return false;
00087     }
00088 }
