00055 inline
00056 AbstractReplacementPolicy::AbstractReplacementPolicy(Index num_sets,
00057                                                      Index assoc)
00058 {
00059     m_num_sets = num_sets;
00060     m_assoc = assoc;
00061     m_last_ref_ptr = new Tick*[m_num_sets];
00062     for(unsigned i = 0; i < m_num_sets; i++){
00063         m_last_ref_ptr[i] = new Tick[m_assoc];
00064         for(unsigned j = 0; j < m_assoc; j++){
00065             m_last_ref_ptr[i][j] = 0;
00066         }
00067     }
00068 }

00070 inline
00071 AbstractReplacementPolicy::~AbstractReplacementPolicy()
00072 {
00073     if (m_last_ref_ptr != NULL){
00074         for (unsigned i = 0; i < m_num_sets; i++){
00075             if (m_last_ref_ptr[i] != NULL){
00076                 delete[] m_last_ref_ptr[i];
00077             }
00078         }
00079         delete[] m_last_ref_ptr;
00080     }
00081 }
