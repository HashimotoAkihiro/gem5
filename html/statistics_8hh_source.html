02312

02314             vresult.resize(1);

02321         } else if (rvec.size() == 1) {
02322             size_type size = lvec.size();
02323             vresult.resize(size);
02324             for (off_type i = 0; i < size; ++i)

02327             size_type size = rvec.size();
02328             vresult.resize(size);

02353             for (off_type i = 0; i < size(); ++i) {
02354                 lsum += lvec[i];
02355                 rsum += rvec[i];
