00098     populate(uint64_t e, EntryType t = sun4u)
00099     {
00100         entry = e;
00101         type = t;
00102         populated = true;
00103 
00104         // If we get a sun4v format TTE, turn it into a sun4u
00105         if (type == sun4u)
00106             entry4u = entry;
00107         else {
00108             entry4u = 0;
00109             entry4u |= mbits(entry,63,63);         // valid
00110             entry4u |= bits(entry,1,0) << 61;      // size[1:0]
00111             entry4u |= bits(entry,62,62) << 60;    // nfo
00112             entry4u |= bits(entry,12,12) << 59;    // ie
00113             entry4u |= bits(entry,2,2) << 48;      // size[2]
00114             entry4u |= mbits(entry,39,13);         // paddr
00115             entry4u |= bits(entry,61,61) << 6;;    // locked
00116             entry4u |= bits(entry,10,10) << 5;     // cp
00117             entry4u |= bits(entry,9,9) << 4;       // cv
00118             entry4u |= bits(entry,11,11) << 3;     // e
00119             entry4u |= bits(entry,8,8) << 2;       // p
00120             entry4u |= bits(entry,6,6) << 1;       // w
00121         }
00122     }

00233     TlbEntry(Addr asn, Addr vaddr, Addr paddr)
00234     {
00235         uint64_t entry = 0;
00236         entry |= 1ULL << 1; // Writable
00237         entry |= 0ULL << 2; // Available in nonpriveleged mode
00238         entry |= 0ULL << 3; // No side effects
00239         entry |= 1ULL << 4; // Virtually cachable
00240         entry |= 1ULL << 5; // Physically cachable
00241         entry |= 0ULL << 6; // Not locked
00242         entry |= mbits(paddr, 39, 13); // Physical address
00243         entry |= 0ULL << 48; // size = 8k
00244         entry |= 0uLL << 59; // Endianness not inverted
00245         entry |= 0ULL << 60; // Not no fault only
00246         entry |= 0ULL << 61; // size = 8k
00247         entry |= 1ULL << 63; // valid
00248         pte = PageTableEntry(entry);
00249 
00250         range.va = vaddr;
00251         range.size = 8*(1<<10);
00252         range.contextId = asn;
00253         range.partitionId = 0;
00254         range.real = false;
00255 
00256         valid = true;
