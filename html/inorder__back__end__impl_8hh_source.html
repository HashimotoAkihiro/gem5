00086 void
00087 InorderBackEnd<Impl>::checkInterrupts()
00088 {
00089     //Check if there are any outstanding interrupts
00090     //Handle the interrupts
00091     int ipl = 0;
00092     int summary = 0;
00093 
00094 
00095     if (thread->readMiscRegNoEffect(IPR_ASTRR))
00096         panic("asynchronous traps not implemented\n");
00097 
00098     if (thread->readMiscRegNoEffect(IPR_SIRR)) {
00099         for (int i = INTLEVEL_SOFTWARE_MIN;
00100              i < INTLEVEL_SOFTWARE_MAX; i++) {
00101             if (thread->readMiscRegNoEffect(IPR_SIRR) & (ULL(1) << i)) {
00102                 // See table 4-19 of the 21164 hardware reference
00103                 ipl = (i - INTLEVEL_SOFTWARE_MIN) + 1;
00104                 summary |= (ULL(1) << i);
00105             }
00106         }
00107     }
00108 
00109     uint64_t interrupts = cpu->intr_status();
00110 
00111     if (interrupts) {
00112         for (int i = INTLEVEL_EXTERNAL_MIN;
00113              i < INTLEVEL_EXTERNAL_MAX; i++) {
00114             if (interrupts & (ULL(1) << i)) {
00115                 // See table 4-19 of the 21164 hardware reference
00116                 ipl = i;
00117                 summary |= (ULL(1) << i);
00118             }
00119         }
00120     }
00121 
00122     if (ipl && ipl > thread->readMiscRegNoEffect(IPR_IPLR)) {
00123         thread->noSquashFromTC = true;
00124 
00125         thread->setMiscRegNoEffect(IPR_ISR, summary);
00126         thread->setMiscRegNoEffect(IPR_INTID, ipl);
00127         Fault(new InterruptFault)->invoke(xc);
00128         DPRINTF(Flow, "Interrupt! IPLR=%d ipl=%d summary=%x\n",
00129                 thread->readMiscRegNoEffect(IPR_IPLR), ipl, summary);
00130 
00131         // May need to go 1 inst prior
00132         squashPending = true;
00133 
00134         thread->noSquashFromTC = false;
00135 
00136         setSquashInfoFromXC();
00137     }

00142 InorderBackEnd<Impl>::tick()
00143 {
00144     // Squash due to an external source
00145     // Not sure if this or an interrupt has higher priority
00146     if (squashPending) {
00147         squash(squashSeqNum, squashNextPC);
00148         return;
00149     }
00150 
00151     // if (interrupt) then set thread PC, stall front end, record that
00152     // I'm waiting for it to drain.  (for now just squash)
00153     if (FullSystem && (interruptBlocked || cpu->checkInterrupts(tc))) {
00154         if (!robEmpty()) {
00155             interruptBlocked = true;
00156         //AlphaDep
00157         } else if (robEmpty() && (PC & 0x3)) {
00158             // Will need to let the front end continue a bit until
00159             // we're out of pal mode.  Hopefully we never get into an
00160             // infinite loop...
00161             interruptBlocked = false;
00162         } else {
00163             interruptBlocked = false;
00164             checkInterrupts();
00165             return;
00166         }
00167     }
00168 
00169     if (status != DcacheMissLoadStall &&
00170         status != DcacheMissStoreStall) {
00171         for (int i = 0; i < width && (*instsAdded) < width; ++i) {
00172             DynInstPtr inst = frontEnd->getInst();
00173 
00174             if (!inst)
00175                 break;
00176 
00177             instList.push_back(inst);
00178 
00179             (*instsAdded)++;
00180         }
00181 
00182         if (faultFromFetch && robEmpty() && frontEnd->isEmpty()) {
00183             handleFault();
00184         } else {
00185             executeInsts();
00186         }
00187     }

00194     bool completed_last_inst = true;
00195     int insts_to_execute = *instsToExecute;
00196     int freed_regs = 0;
00197 
00198     while (insts_to_execute > 0) {
00199         assert(!instList.empty());
00200         DynInstPtr inst = instList.front();
00201 
00202         commitPC = inst->readPC();
00203 
00204         thread->setPC(commitPC);
00205         thread->setNextPC(inst->readNextPC());
00206 
00207         if (FullSystem) {
00208             int count = 0;
00209             Addr oldpc;
00210             do {
00211                 if (count == 0)
00212                     assert(!thread->noSquashFromTC && !thread->trapPending);
00213                 oldpc = thread->readPC();
00214                 cpu->system->pcEventQueue.service(
00215                     thread->getXCProxy());
00216                 count++;
00217             } while (oldpc != thread->readPC());
00218             if (count > 1) {
00219                 DPRINTF(IBE, "PC skip function event, stopping commit\n");
00220                 completed_last_inst = false;
00221                 squashPending = true;
00222                 break;
00223             }
00224         }
00225 
00226         Fault inst_fault = NoFault;
00227 
00228         if (status == DcacheMissComplete) {
00229             DPRINTF(IBE, "Completing inst [sn:%lli]\n", inst->seqNum);
00230             status = Running;
00231         } else if (inst->isMemRef() && status != DcacheMissComplete &&
00232             (!inst->isDataPrefetch() && !inst->isInstPrefetch())) {
00233             DPRINTF(IBE, "Initiating mem op inst [sn:%lli] PC: %#x\n",
00234                     inst->seqNum, inst->readPC());
00235 
00236             cacheCompletionEvent.inst = inst;
00237             inst_fault = inst->initiateAcc();
00238             if (inst_fault == NoFault &&
00239                 status != DcacheMissLoadStall &&
00240                 status != DcacheMissStoreStall) {
00241                 inst_fault = inst->completeAcc();
00242             }
00243             ++thread->funcExeInst;
00244         } else {
00245             DPRINTF(IBE, "Executing inst [sn:%lli] PC: %#x\n",
00246                     inst->seqNum, inst->readPC());
00247             inst_fault = inst->execute();
00248             ++thread->funcExeInst;
00249         }
00250 
00251         // Will need to be able to break this loop in case the load
00252         // misses.  Split access/complete ops would be useful here
00253         // with writeback events.
00254         if (status == DcacheMissLoadStall) {
00255             *instsToExecute = insts_to_execute;
00256 
00257             completed_last_inst = false;
00258             break;
00259         } else if (status == DcacheMissStoreStall) {
00260             // Figure out how to fix this hack.  Probably have DcacheMissLoad
00261             // vs DcacheMissStore.
00262             *instsToExecute = insts_to_execute;
00263             completed_last_inst = false;
00264 /*
00265             instList.pop_front();
00266             --insts_to_execute;
00267             if (inst->traceData) {
00268                 inst->traceData->finalize();
00269             }
00270 */
00271 
00272             // Don't really need to stop for a store stall as long as
00273             // the memory system is able to handle store forwarding
00274             // and such.  Breaking out might help avoid the cache
00275             // interface becoming blocked.
00276             break;
00277         }
00278 
00279         inst->setExecuted();
00280         inst->setResultReady();
00281         inst->setCanCommit();
00282 
00283         instList.pop_front();
00284 
00285         --insts_to_execute;
00286         --(*instsToExecute);
00287 
00288         if (inst->traceData) {
00289             inst->traceData->finalize();
00290             inst->traceData = NULL;
00291         }
00292 
00293         if (inst_fault != NoFault) {
00294             DPRINTF(IBE, "Inst [sn:%lli] PC %#x has a fault\n",
00295                     inst->seqNum, inst->readPC());
00296 
00297             assert(!thread->noSquashFromTC);
00298 
00299             thread->noSquashFromTC = true;
00300 
00301             // Consider holding onto the trap and waiting until the trap event
00302             // happens for this to be executed.
00303             inst_fault->invoke(xc);
00304 
00305             // Exit state update mode to avoid accidental updating.
00306             thread->noSquashFromTC = false;
00307 
00308             squashPending = true;
00309 
00310             completed_last_inst = false;
00311             break;
00312         }
00313 
00314         for (int i = 0; i < inst->numDestRegs(); ++i) {
00315             renameTable[inst->destRegIdx(i)] = inst;
00316             thread->renameTable[inst->destRegIdx(i)] = inst;
00317             ++freed_regs;
00318         }
00319 
00320         inst->clearDependents();
00321 
00322         comm->access(0)->doneSeqNum = inst->seqNum;
00323 
00324         if (inst->mispredicted()) {
00325             squash(inst->seqNum, inst->readNextPC());
00326 
00327             thread->setNextPC(inst->readNextPC());
00328 
00329             break;
00330         } else if (squashPending) {
00331             // Something external happened that caused the CPU to squash.
00332             // Break out of commit and handle the squash next cycle.
00333             break;
00334         }
00335         // If it didn't mispredict, then it executed fine.  Send back its
00336         // registers and BP info?  What about insts that may still have
00337         // latency, like loads?  Probably can send back the information after
00338         // it is completed.
00339 
00340         // keep an instruction count
00341         cpu->numInst++;
00342         thread->numInsts++;
00343     }
00344 
00345     frontEnd->addFreeRegs(freed_regs);
00346 
00347     assert(insts_to_execute >= 0);
00348 
00349     // Should only advance this if I have executed all instructions.
00350     if (insts_to_execute == 0) {
00351         numInstsToWB.advance();
00352     }
00353 
00354     // Should I set the PC to the next PC here?  What do I set next PC to?
00355     if (completed_last_inst) {
00356         thread->setPC(thread->readNextPC());
00357         thread->setNextPC(thread->readPC() + sizeof(MachInst));
00358     }
00359 
00360     if (squashPending) {
00361         setSquashInfoFromXC();
00362     }

00388 void
00389 InorderBackEnd<Impl>::squash(const InstSeqNum &squash_num, const Addr &next_PC)
00390 {
00391     DPRINTF(IBE, "Squashing from [sn:%lli], setting PC to %#x\n",
00392             squash_num, next_PC);
00393 
00394     InstListIt squash_it = --(instList.end());
00395 
00396     int freed_regs = 0;
00397 
00398     while (!instList.empty() && (*squash_it)->seqNum > squash_num) {
00399         DynInstPtr inst = *squash_it;
00400 
00401         DPRINTF(IBE, "Squashing instruction PC %#x, [sn:%lli].\n",
00402                 inst->readPC(),
00403                 inst->seqNum);
00404 
00405         // May cause problems with misc regs
00406         freed_regs+= inst->numDestRegs();
00407         inst->clearDependents();
00408         squash_it--;
00409         instList.pop_back();
00410     }
00411 
00412     frontEnd->addFreeRegs(freed_regs);
00413 
00414     for (int i = 0; i < latency+1; ++i) {
00415         numInstsToWB.advance();
00416     }
00417 
00418     squashPending = false;
00419 
00420     // Probably want to make sure that this squash is the one that set the
00421     // thread into noSquashFromTC mode.
00422     thread->noSquashFromTC = false;
00423 
00424     // Tell front end to squash, reset PC to new one.
00425     frontEnd->squash(squash_num, next_PC);
00426 
00427     faultFromFetch = NULL;
00428 }
