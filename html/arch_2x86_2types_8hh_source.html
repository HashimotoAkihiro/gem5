00131     struct ExtMachInst
00132     {
00133         //Prefixes
00134         LegacyPrefixVector legacy;
00135         Rex rex;
00136         //This holds all of the bytes of the opcode
00137         struct
00138         {
00139             //The number of bytes in this opcode. Right now, we ignore that
00140             //this can be 3 in some cases
00141             uint8_t num;
00142             //The first byte detected in a 2+ byte opcode. Should be 0xF0.
00143             uint8_t prefixA;
00144             //The second byte detected in a 3+ byte opcode. Could be 0x38-0x3F
00145             //for some SSE instructions. 3dNow! instructions are handled as
00146             //two byte opcodes and then split out further by the immediate
00147             //byte.
00148             uint8_t prefixB;
00149             //The main opcode byte. The highest addressed byte in the opcode.
00150             Opcode op;
00151         } opcode;
00152         //Modifier bytes
00153         ModRM modRM;
00154         Sib sib;
00155         //Immediate fields
00156         uint64_t immediate;
00157         uint64_t displacement;
