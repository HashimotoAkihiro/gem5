00080     {
00081         int64_t midRes = sub ? (op1 - op2) : (op1 + op2);
00082         if (bits(midRes, width) != bits(midRes, width - 1)) {
00083             if (midRes > 0)
00084                 res = (LL(1) << (width - 1)) - 1;
00085             else

00191     static inline uint32_t
00192     cpsrWriteByInstr(CPSR cpsr, uint32_t val, SCR scr, NSACR nsacr,
00193             uint8_t byteMask, bool affectState, bool nmfi, ThreadContext *tc)
00194     {
00195         bool privileged   = (cpsr.mode != MODE_USER);
00196         bool haveVirt     = ArmSystem::haveVirtualization(tc);
00197         bool haveSecurity = ArmSystem::haveSecurity(tc);
00198         bool isSecure     = inSecureState(scr, cpsr) || !haveSecurity;
00199 
00200         uint32_t bitMask = 0;
00201 
00202         if (bits(byteMask, 3)) {
00203             unsigned lowIdx = affectState ? 24 : 27;
00204             bitMask = bitMask | mask(31, lowIdx);
00205         }
00206         if (bits(byteMask, 2)) {
00207             bitMask = bitMask | mask(19, 16);
00208         }
00209         if (bits(byteMask, 1)) {
00210             unsigned highIdx = affectState ? 15 : 9;
00211             unsigned lowIdx = (privileged && (isSecure || scr.aw || haveVirt))
00212                             ? 8 : 9;
00213             bitMask = bitMask | mask(highIdx, lowIdx);
00214         }
00215         if (bits(byteMask, 0)) {
00216             if (privileged) {
00217                 bitMask |= 1 << 7;
00218                 if ( (!nmfi || !((val >> 6) & 0x1)) &&
00219                      (isSecure || scr.fw || haveVirt) ) {
00220                     bitMask |= 1 << 6;
00221                 }
00222                 // Now check the new mode is allowed
00223                 OperatingMode newMode = (OperatingMode) (val & mask(5));
00224                 OperatingMode oldMode = (OperatingMode)(uint32_t)cpsr.mode;
00225                 if (!badMode(newMode)) {
00226                     bool validModeChange = true;
00227                     // Check for attempts to enter modes only permitted in
00228                     // Secure state from Non-secure state. These are Monitor
00229                     // mode ('10110'), and FIQ mode ('10001') if the Security
00230                     // Extensions have reserved it.
00231                     if (!isSecure && newMode == MODE_MON)
00232                         validModeChange = false;
00233                     if (!isSecure && newMode == MODE_FIQ && nsacr.rfr == '1')
00234                         validModeChange = false;
00235                     // There is no Hyp mode ('11010') in Secure state, so that
00236                     // is UNPREDICTABLE
00237                     if (scr.ns == '0' && newMode == MODE_HYP)
00238                         validModeChange = false;
00239                     // Cannot move into Hyp mode directly from a Non-secure
00240                     // PL1 mode
00241                     if (!isSecure && oldMode != MODE_HYP && newMode == MODE_HYP)
00242                         validModeChange = false;
00243                     // Cannot move out of Hyp mode with this function except
00244                     // on an exception return
00245                     if (oldMode == MODE_HYP && newMode != MODE_HYP && !affectState)
00246                         validModeChange = false;
00247                     // Must not change to 64 bit when running in 32 bit mode
00248                     if (!opModeIs64(oldMode) && opModeIs64(newMode))
00249                         validModeChange = false;
00250 
00251                     // If we passed all of the above then set the bit mask to
00252                     // copy the mode accross
00253                     if (validModeChange) {
00254                         bitMask = bitMask | mask(5);
00255                     } else {
00256                         warn_once("Illegal change to CPSR mode attempted\n");
00257                     }
00258                 } else {
00259                     warn_once("Ignoring write of bad mode to CPSR.\n");
00260                 }
00261             }
00262             if (affectState)
00263                 bitMask = bitMask | (1 << 5);
00264         }
00265 
00266         return ((uint32_t)cpsr & ~bitMask) | (val & bitMask);
00267     }

00315     static inline T
00316     cSwap(T val, bool big)
00317     {
00318         const unsigned count = sizeof(T) / sizeof(E);
00319         union {
00320             T tVal;
00321             E eVals[count];
00322         } conv;
00323         conv.tVal = htog(val);
00324         if (big) {
00325             for (unsigned i = 0; i < count; i++) {
00326                 conv.eVals[i] = gtobe(conv.eVals[i]);
00327             }
00328         } else {
00329             for (unsigned i = 0; i < count; i++) {
00330                 conv.eVals[i] = gtole(conv.eVals[i]);
00331             }
00332         }
00333         return gtoh(conv.tVal);
