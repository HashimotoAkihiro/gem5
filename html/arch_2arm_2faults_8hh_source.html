00209 class ArmFaultVals : public ArmFault
00210 {
00211   protected:
00212     static FaultVals vals;
00213 
00214   public:
00215     ArmFaultVals<T>(ExtMachInst _machInst = 0, uint32_t _iss = 0) :
00216         ArmFault(_machInst, _iss) {}
00217     FaultName name() const { return vals.name; }
00218     FaultStat & countStat() { return vals.count; }
00219     FaultOffset offset(ThreadContext *tc);
00220 
00221     FaultOffset
00222     offset64()
00223     {
00224         if (toEL == fromEL) {
00225             if (opModeIsT(fromMode))
00226                 return vals.currELTOffset;
00227             return vals.currELHOffset;
00228         } else {
00229             if (from64)
00230                 return vals.lowerEL64Offset;
00231             return vals.lowerEL32Offset;
00232         }
00233     }
00234 
00235     OperatingMode nextMode() { return vals.nextMode; }
00236     virtual bool routeToMonitor(ThreadContext *tc) const { return false; }
00237     uint8_t armPcOffset(bool isHyp)   { return isHyp ? vals.armPcElrOffset
00238                                                      : vals.armPcOffset; }
00239     uint8_t thumbPcOffset(bool isHyp) { return isHyp ? vals.thumbPcElrOffset
00240                                                      : vals.thumbPcOffset; }
00241     uint8_t armPcElrOffset() { return vals.armPcElrOffset; }
00242     uint8_t thumbPcElrOffset() { return vals.thumbPcElrOffset; }
00243     virtual bool abortDisable(ThreadContext* tc) { return vals.abortDisable; }
00244     virtual bool fiqDisable(ThreadContext* tc) { return vals.fiqDisable; }
00245     virtual ExceptionClass ec(ThreadContext *tc) const { return vals.ec; }
00246     virtual uint32_t iss() const { return issRaw; }
